 59    def convert_openGl(self, matrix):
 60
 61        matrix_copy = matrix.copy()
 62        C = mathutils.Matrix(((1,0,0,0),
 63                      (0,0,1,0),
 64                      (0,-1,0,0),
 65                      (0,0,0,1)))
 66        C_inv = C.transposed() 
 67
 68        return C @ matrix_copy @ C_inv

 70    def should_export_object(self, obj):
 71        """
 72        Determines whether an object should be exported based on user settings.
 73        """
 74        if not obj:
 75            return False
 76
 77        # Controlla il tipo di oggetto
 78        if obj.type == 'MESH' and not self.use_mesh:
 79            return False
 80        if obj.type == 'LIGHT' and not self.use_light:
 81            return False
 82
 83        return True

 85    def write_node_recursive(self, file, obj):
 86        """
 87        Writes a node and its children recursively into the OVO file.
 88        """
 89        if obj in self.processed_objects:
 90            return
 91
 92        self.processed_objects.add(obj)
 93
 94        # Process children
 95        valid_children = []
 96        for child in obj.children:
 97            if child not in self.processed_objects:
 98                if ((child.type == 'MESH' and self.use_mesh) or
 99                        (child.type == 'LIGHT' and self.use_light) or
100                        (child.type not in {'MESH', 'LIGHT'})):
101                    valid_children.append(child)
102
103        num_children = len(valid_children)
104
105        if obj.type == 'MESH':
106            category = "MESH"
107        elif obj.type == 'LIGHT':
108            category = "LIGHT"
109        else:
110            category = "MESH"
111
112        log(f"[OVOExporter] Processing: {obj.name}", category=category, indent=2)
113        log(f"- Type: {obj.type}", category=category, indent=2)
114        log(f"- Children: {num_children}", category=category, indent=2)
115        log(f"- Should export: {self.should_export_object(obj)}", category=category, indent=2)
116
117        # Process materials
118        if obj.type == 'MESH' and self.should_export_object(obj):
119            for material_slot in obj.material_slots:
120                material = material_slot.material
121                if material and material not in self.processed_objects:
122                    print(f"      - Writing material: {material.name}")
123                    self.write_material_chunk(file, material)
124                    self.processed_objects.add(material)
125
126        if ((obj.type == 'MESH' and self.use_mesh) or
127                (obj.type == 'LIGHT' and self.use_light) or
128                (obj.type not in {'MESH', 'LIGHT'})):
129            if obj.type == 'MESH':
130                print(f"      - Writing mesh chunk")
131                self.write_mesh_chunk(file, obj, num_children)
132            elif obj.type == 'LIGHT':
133                print(f"      - Writing light chunk")
134                self.write_light_chunk(file, obj, num_children)
135            else:
136                print(f"      - Writing node chunk")
137                self.write_node_chunk(file, obj, num_children)
138
139        # Process children recursively
140        for child in valid_children:
141            self.write_node_recursive(file, child)

157    def write_material_chunk(self, file, material):
158        """
159        Writes a material chunk to the OVO file.
160
161        Args:
162            file: Output file object
163            material: Blender material to export
164        """
165
166        log(f"[OVOExporter.write_material_chunk] Processing material: {material.name}", category="MATERIAL", indent=2)
167        chunk_data = b''  # byte chunk, not string
168
169        # Material name
170        chunk_data += self.packer.pack_string(material.name)
171
172        # Default values
173        emission_color = (0, 0, 0)
174        base_color_rgb = (0.8, 0.8, 0.8)
175        alpha = 1.0
176        roughness = 0.5
177        metallic = 0.0
178
179        # Default texture file values
180        albedo_texture = "[none]"
181        normal_texture = "[none]"
182        roughness_texture = "[none]"
183        metallic_texture = "[none]"
184        height_texture = "[none]"
185
186        # Extract material properties
187        if material.use_nodes and material.node_tree:
188            principled = material.node_tree.nodes.get('Principled BSDF')
189            emission_node = material.node_tree.nodes.get('Emission')
190
191            # Emission conversion (from Blender RGBA to RGB for OVO)
192            if emission_node:
193                emission = emission_node.inputs[0].default_value
194                emission_color = emission[:3] if len(emission) > 2 else (0, 0, 0)
195                log(f"- Emission: ({emission_color[0]:.3f}, {emission_color[1]:.3f}, {emission_color[2]:.3f})",category="MATERIAL", indent=3)
196
197            if principled:
198                log("Found Principled BSDF node", category="MATERIAL", indent=3)
199
200                # Base Color and related texture
201                base_color_input = principled.inputs.get('Base Color')
202                if base_color_input:
203                    if base_color_input.is_linked:
204                        log("- Base Color has linked texture", category="MATERIAL", indent=3)
205                        albedo_texture = self.texture_manager.trace_to_image_node(base_color_input, isAlbedo=True)
206                        if albedo_texture != "[none]":
207                            log(f"- Albedo texture: '{albedo_texture}'", category="MATERIAL", indent=3)
208                    else:
209                        base_color = base_color_input.default_value
210                        base_color_rgb = base_color[:3] if len(base_color) > 2 else (0.8, 0.8, 0.8)
211                        alpha = base_color[3] if len(base_color) > 3 else 1.0
212                        log(f"- Base Color: ({base_color_rgb[0]:.3f}, {base_color_rgb[1]:.3f}, {base_color_rgb[2]:.3f})", category="MATERIAL", indent=3)
213                        log(f"- Alpha: {alpha:.3f}", category="MATERIAL", indent=3)
214
215                # Material properties
216                roughness = principled.inputs['Roughness'].default_value
217                metallic = principled.inputs['Metallic'].default_value
218                log(f"- Roughness: {roughness:.3f}", category="MATERIAL", indent=3)
219                log(f"- Metallic: {metallic:.3f}", category="MATERIAL", indent=3)
220
221                # Other textures
222                normal_input = principled.inputs.get('Normal')
223                if normal_input:
224                    normal_texture = self.texture_manager.trace_to_image_node(normal_input)
225                    if normal_texture != "[none]":
226                        log(f"- Normal texture: '{normal_texture}'", category="MATERIAL", indent=3)
227
228                roughness_input = principled.inputs.get('Roughness')
229                if roughness_input:
230                    roughness_texture = self.texture_manager.trace_to_image_node(roughness_input)
231                    if roughness_texture != "[none]":
232                        log(f"- Roughness texture: '{roughness_texture}'", category="MATERIAL", indent=3)
233
234                metallic_input = principled.inputs.get('Metallic')
235                if metallic_input:
236                    metallic_texture = self.texture_manager.trace_to_image_node(metallic_input)
237                    if metallic_texture != "[none]":
238                        log(f"- Metallic texture: '{metallic_texture}'", category="MATERIAL", indent=3)
239
240                height_input = principled.inputs.get('Height')
241                if height_input:
242                    height_texture = self.texture_manager.trace_to_image_node(height_input)
243                    if height_texture != "[none]":
244                        log(f"- Height texture: '{height_texture}'", category="MATERIAL", indent=3)
245        else:
246            log("- Material has no nodes, using default values", category="MATERIAL", indent=3)
247
248        # Write binary data to chunk
249        chunk_data += struct.pack('3f', *emission_color)
250        chunk_data += struct.pack('3f', *base_color_rgb)
251        chunk_data += struct.pack('f', roughness)
252        chunk_data += struct.pack('f', metallic)
253        chunk_data += struct.pack('f', alpha)
254
255        # Write texture paths
256        chunk_data += self.packer.pack_string(albedo_texture)
257        chunk_data += self.packer.pack_string(normal_texture)
258        chunk_data += self.packer.pack_string(height_texture)
259        chunk_data += self.packer.pack_string(roughness_texture)
260        chunk_data += self.packer.pack_string(metallic_texture)
261
262        # Write chunk header and chunk itself to file
263        log("Writing material chunk to file", category="MATERIAL", indent=3)
264        self.packer.write_chunk_header(file, ChunkType.MATERIAL, len(chunk_data))
265        file.write(chunk_data)
266
267        log(f"[OVOExporter.write_material_chunk] Completed: '{material.name}'", category="MATERIAL", indent=2)

274        Writes a basic node chunk for objects that aren't mesh or light.

277            file: Output file object
278            obj: Blender object to export
279            num_children: Number of children for this node
280        """
281        log(f"[OVOExporter.write_node_chunk] Processing node: '{obj.name}'", category="NODE", indent=2)
282        chunk_data = b''  # binary
283
284        # Node name
285        chunk_data += self.packer.pack_string(obj.name)
286
287
288        # In write_mesh_chunk
289        if obj.parent:
290            local_bl = obj.matrix_local  
291        else:
292            local_bl = obj.matrix_world
293
294        # Pack and convert matrix

296        chunk_data += self.packer.pack_matrix(final_matrix)
297        log("- Matrix transformed and packed", category="NODE", indent=3)
298
299        # Number of children
300        chunk_data += struct.pack('I', num_children)
301        log(f"- Children count: {num_children}", category="NODE", indent=3)
302
303        # Target node
304        chunk_data += self.packer.pack_string("[none]")
305
306        # Debug additional information
307        if num_children > 0:
308            children_names = [child.name for child in obj.children
309                              if child not in self.processed_objects
310                              and self.should_export_object(child)]
311            if children_names:
312                joined_names = ', '.join(children_names)
313                log(f"- Child nodes: {joined_names}", category="NODE", indent=3)
314
315        # Write the chunk
316        # Write the chunk
317        log("- Writing node chunk to file", category="NODE", indent=3)
318        self.packer.write_chunk_header(file, ChunkType.NODE, len(chunk_data))
319        file.write(chunk_data)

325    def write_mesh_chunk(self, file, obj, num_children):
326        """
327        Writes a mesh chunk to the OVO file.
328
329        Args:
330            file: Output file object
331            obj: Blender mesh object to export
332            num_children: Number of children for this node
333        """
334        chunk_data = b''
335
336        # Mesh name
337        log(f"[OVOExporter.write_mesh_chunk] Processing mesh: '{obj.name}'", category="MESH", indent=2)
338        chunk_data += self.packer.pack_string(obj.name)
339
340        # In write_mesh_chunk
341        if obj.parent:
342            local_bl = obj.matrix_local  
343        else:
344            local_bl = obj.matrix_world
345
346        # Save matrix without additional conversions
347        final_matrix = self.convert_openGl(local_bl)
348        chunk_data += self.packer.pack_matrix(final_matrix)
349
350        # Children and material data
351        chunk_data += struct.pack('I', num_children)
352        chunk_data += self.packer.pack_string("[none]")
353        chunk_data += struct.pack('B', 0)
354
355                
356        # Material assignment
357        if obj.material_slots and obj.material_slots[0].material:
358            material_name = obj.material_slots[0].material.name
359            chunk_data += self.packer.pack_string(material_name)
360            log(f"- Material: '{material_name}'", category="MESH", indent=3)
361        else:
362            chunk_data += self.packer.pack_string("[none]")
363            log("- No material assigned", category="MESH", indent=3)
364
365        # Get mesh data from evaluated object
366        log("- Getting mesh data", category="MESH", indent=3)
367        depsgraph = bpy.context.evaluated_depsgraph_get()
368        obj_eval = obj.evaluated_get(depsgraph)
369        mesh = obj_eval.to_mesh()
370
371        # Tangent
372        try:
373            mesh.calc_tangents()
374            loop_tangent = [l.tangent.copy()  for l in mesh.loops]
375            loop_sign    = [l.bitangent_sign  for l in mesh.loops]
376        except RuntimeError:
377            log("- Tangent calculation failed (n-gons detected), using safe method", category="MESH", indent=3)
378            loop_tangent, loop_sign = self.mesh_manager.safe_calc_tangents(mesh)
379
380        loop_tangent = [l.tangent.copy()      for l in mesh.loops]
381        loop_sign    = [l.bitangent_sign      for l in mesh.loops]
382        poly_lstart  = [p.loop_start          for p in mesh.polygons]
383
384        # Create two BMesh instances: original and one to be triangulated
385        import bmesh
386        original_bm = bmesh.new()
387        original_bm.from_mesh(mesh)
388
389        triangulated_bm = bmesh.new()
390        triangulated_bm.from_mesh(mesh)
391        bmesh.ops.triangulate(triangulated_bm, faces=triangulated_bm.faces)
392
393        # Ensure lookup table are updated
394        original_bm.faces.ensure_lookup_table()
395        original_bm.verts.ensure_lookup_table()
396        triangulated_bm.faces.ensure_lookup_table()
397        triangulated_bm.verts.ensure_lookup_table()
398
399        # Get UV layer
400        uv_layer = original_bm.loops.layers.uv.active
401        if uv_layer:
402            log(f"- UV layer found: '{mesh.uv_layers.active.name if mesh.uv_layers.active else 'default'}'", category="MESH", indent=3)
403        else:
404            log("- WARNING: No UV layer found", category="MESH", indent=3)
405
406        radius, min_box, max_box = self.mesh_manager.get_box_radius(mesh.verices)
407
408        # Write bounding box information
409        chunk_data += struct.pack('f', radius)
410        chunk_data += self.packer.pack_vector3(min_box)
411        chunk_data += self.packer.pack_vector3(max_box)
412
413        # Process physics data
414        log("- Processing physics data", category="MESH", indent=3)
415        chunk_data = self.physics_manager.write_physics_data(obj, chunk_data)
416        lod_manager = OVOLodManager()
417
418        # Check face count to determine if we need multi-LOD
419        should_multi_lod = lod_manager.should_generate_multi_lod(obj)
420
421        # Vertex data
422        vertex_count = 0
423        face_count = 0
424        vertices_data = []
425        face_indices = []
426
427        if should_multi_lod:
428            # Generate LOD meshes
429            log("- Generating multiple LODs for high-poly mesh", category="MESH", indent=3)
430            lod_meshes = lod_manager.generate_lod_meshes(obj)
431
432            # Write LOD data
433            chunk_data = self.write_lod_data(obj, chunk_data, lod_meshes)
434
435            # Clean up LOD meshes
436            lod_manager.cleanup_lod_meshes(lod_meshes)
437        else:
438
439            # Original single LOD code path single LOD
440            log("- LOD count: 1 (single LOD)", category="MESH", indent=3)
441            chunk_data += struct.pack('I', 1)
442
443            vertices_data = []
444            face_indices = []
445            face_vertex_map = {}
446
447            for face_idx, face in enumerate(original_bm.faces):
448                # Calculate face normals (not interpolated)
449                face_normal = face.normal.normalized()
450                
451                
452                # Iterate on faces to extract vertices data
453                for loop_idx, loop in enumerate(face.loops):
454                    vert_idx = loop.vert.index
455                    pos      = loop.vert.co
456                    uv       = loop[uv_layer].uv if uv_layer else mathutils.Vector((0.0, 0.0))
457
458                    mesh_loop_index = poly_lstart[face_idx] + loop_idx
459                    bl_tangent      = loop_tangent[mesh_loop_index]
460                    sign            = loop_sign   [mesh_loop_index]
461
462
463                    # Transform coordinates
464                    transformed_pos  = mathutils.Vector((pos.x,         pos.z,        -pos.y))
465                    transformed_norm = mathutils.Vector((face_normal.x, face_normal.z, -face_normal.y)).normalized()
466                    transformed_tan  = mathutils.Vector((bl_tangent.x,  bl_tangent.z, -bl_tangent.y)).normalized()
467
468                    # Add to lists
469                    vertex_idx = len(vertices_data)
470                    vertices_data.append((transformed_pos, transformed_norm, uv,
471                                        transformed_tan, sign))
472
473                    face_vertex_map[(face_idx, vert_idx)] = vertex_idx
474
475            
476            # Indices for triangulated faces 
477            face_indices = []
478
479            log("- Manual triangulate started", category="MESH", indent=3)
480            face_count = 0
481            for face_idx, face in enumerate(original_bm.faces):
482                # If we have a quad (4 vertices)
483                if len(face.verts) == 4:
484                
485                    v_indices = [face_vertex_map.get((face_idx, v.index), 0) for v in face.verts]
486                    face_indices.extend([v_indices[0], v_indices[1], v_indices[2]])
487                    face_indices.extend([v_indices[0], v_indices[2], v_indices[3]])
488                    face_count += 2
489
490                elif len(face.verts) == 3:
491                    # If we already have a tris 
492                    v_indices = [face_vertex_map.get((face_idx, v.index), 0) for v in face.verts]
493                    face_indices.extend(v_indices)
494                    face_count += 1
495                else:
496                    # For n-gons faces
497                    v0_idx = face_vertex_map.get((face_idx, face.verts[0].index), 0)
498                    for i in range(1, len(face.verts) - 1):
499                        v1_idx = face_vertex_map.get((face_idx, face.verts[i].index), 0)
500                        v2_idx = face_vertex_map.get((face_idx, face.verts[i+1].index), 0)
501                        face_indices.extend([v0_idx, v1_idx, v2_idx])
502                        face_count += 1
503
504            # write faces and vertices
505            vertex_count = len(vertices_data)
506            log(f"- Vertices: {vertex_count}", category="MESH", indent=3)
507            log(f"- Triangulated faces: {face_count}", category="MESH", indent=3)
508            chunk_data += struct.pack('I', vertex_count)
509            chunk_data += struct.pack('I', face_count)
510
511        # If we have a single lod
512        if not should_multi_lod:
513            log(f"- Writing {vertex_count} vertices", category="MESH", indent=3)
514        
515            # Write vertex data
516            for pos, norm, uv, tan, sign in vertices_data:
517                chunk_data += self.packer.pack_vector3(pos)
518                chunk_data += self.packer.pack_normal(norm)
519                chunk_data += self.packer.pack_uv(uv)
520                chunk_data += self.packer.pack_tangent(tan)
521
522            # Write face data
523            log(f" - Writing {face_count} faces", category="MESH", indent=3)
524            for i in range(0, len(face_indices), 3):
525                if i + 2 < len(face_indices):
526                    for j in range(3):
527                        chunk_data += struct.pack('I', face_indices[i + j])
528
529        # Write the complete mesh chunk
530        log(f"- Writing mesh chunk to file", category="MESH", indent=3)
531        self.packer.write_chunk_header(file, ChunkType.MESH, len(chunk_data))
532        file.write(chunk_data)
533
534        # Cleanup
535        original_bm.free()
536        triangulated_bm.free()
537        obj_eval.to_mesh_clear()
538        log(f"[OVOExporter.write_mesh_chunk] Completed: '{obj.name}'", indent=3)

543    def write_lod_data(self, obj, chunk_data, lod_meshes):
544        """
545        Writes LOD data to the chunk.
546
547        Args:
548            obj: Blender mesh object
549            chunk_data: Current chunk data buffer
550            lod_meshes: List of BMesh objects for each LOD
551
552        Returns:
553            bytes: Updated chunk data with LOD information
554        """
555        # Write the number of LODs
556        lod_count = len(lod_meshes)
557        chunk_data += struct.pack('I', lod_count)
558        log(f"- LOD count: {lod_count}", indent=3)
559
560        # Process each LOD level
561        for lod_index, bm in enumerate(lod_meshes):
562            log(f"- Processing LOD {lod_index + 1}/{lod_count}", indent=3)
563
564            # Convert original mesh to BMesh to calculate tangents
565            temp_mesh = bpy.data.meshes.new(f"temp_lod_{lod_index}")
566            bm.to_mesh(temp_mesh)
567
568            # Tangents
569            try:
570                temp_mesh.calc_tangents()
571                loop_tangent = [l.tangent.copy()  for l in temp_mesh.loops]
572                loop_sign    = [l.bitangent_sign  for l in temp_mesh.loops]
573            except RuntimeError:
574                log(f"- LOD {lod_index + 1}: mesh.calc_tangents() failes: using secure fallback method", indent=3)
575                loop_tangent, loop_sign = self.mesh_manager.safe_calc_tangents(temp_mesh)
576            
577            # Get loop start index for each polygon
578            poly_lstart = [p.loop_start for p in temp_mesh.polygons]
579
580            # Get the UV layer
581            uv_layer = bm.loops.layers.uv.active
582            if uv_layer:
583                log(f"- LOD {lod_index + 1}: UV layer found: '{temp_mesh.uv_layers.active.name if temp_mesh.uv_layers.active else 'default'}'", indent=3)
584            else:
585                log(f"- LOD {lod_index + 1}: WARNING - No UV layer found", indent=3)
586
587            # Vertex data
588            vertices_data = []
589            face_indices = []
590            face_vertex_map = {}
591
592            # Process faces
593            log(f"- LOD {lod_index + 1}: Processando le facce originali", indent=3)
594
595            for face_idx, face in enumerate(bm.faces):
596                
597                face_normal = face.normal.normalized()
598                
599                # For each vertex in this face
600                for loop_idx, loop in enumerate(face.loops):
601                    vert_idx = loop.vert.index
602                    pos = loop.vert.co
603                    uv = loop[uv_layer].uv if uv_layer else mathutils.Vector((0.0, 0.0))
604
605                    # Get loop face index
606                    mesh_loop_index = poly_lstart[face_idx] + loop_idx
607                    if mesh_loop_index < len(loop_tangent):
608                        bl_tangent = loop_tangent[mesh_loop_index]
609                        sign = loop_sign[mesh_loop_index]
610                    else:
611                        # fallback
612                        bl_tangent = mathutils.Vector((1.0, 0.0, 0.0))
613                        sign = 1.0
614
615                    # Transform coordinates for OpenGL
616                    transformed_pos = mathutils.Vector((pos.x, pos.z, -pos.y))
617                    transformed_norm = mathutils.Vector((face_normal.x, face_normal.z, -face_normal.y)).normalized()
618                    transformed_tan = mathutils.Vector((bl_tangent.x, bl_tangent.z, -bl_tangent.y)).normalized()
619
620                    # Create a unique vertex
621                    vertex_idx = len(vertices_data)
622                    vertices_data.append((transformed_pos, transformed_norm, uv, transformed_tan, sign))
623                    face_vertex_map[(face_idx, vert_idx)] = vertex_idx
624
625            # Triangulate faces
626            log(f"- LOD triangulating faces", indent=3)
627            face_count = 0
628            for face_idx, face in enumerate(bm.faces):
629                # If we have a quad (4 vertices)
630                if len(face.verts) == 4:
631                    
632                    v_indices = [face_vertex_map.get((face_idx, v.index), 0) for v in face.verts]
633                    
634                    face_indices.extend([v_indices[0], v_indices[1], v_indices[2]])
635                    face_indices.extend([v_indices[0], v_indices[2], v_indices[3]])
636                    face_count += 2 
637                elif len(face.verts) == 3:
638                    # If we already have a tris
639                    v_indices = [face_vertex_map.get((face_idx, v.index), 0) for v in face.verts]
640                    face_indices.extend(v_indices)
641                    face_count += 1 
642                else:
643                    # For n-gons faces
644                    v0_idx = face_vertex_map.get((face_idx, face.verts[0].index), 0)
645                    for i in range(1, len(face.verts) - 1):
646                        v1_idx = face_vertex_map.get((face_idx, face.verts[i].index), 0)
647                        v2_idx = face_vertex_map.get((face_idx, face.verts[i+1].index), 0)
648                        face_indices.extend([v0_idx, v1_idx, v2_idx])
649                        face_count += 1 
650
651            # Write vertices and faces count
652            vertex_count = len(vertices_data)
653            log(f"- LOD {lod_index + 1}: {vertex_count} vertices, {face_count} faces", indent=3)
654            chunk_data += struct.pack('I', vertex_count)
655            chunk_data += struct.pack('I', face_count)
656
657            # Write vertex data
658            log(f"- LOD {lod_index + 1}: Writing {vertex_count} vertices", indent=3)
659            for pos, norm, uv, tan, sign in vertices_data:
660                chunk_data += self.packer.pack_vector3(pos)
661                chunk_data += self.packer.pack_normal(norm)
662                chunk_data += self.packer.pack_uv(uv)
663                chunk_data += self.packer.pack_tangent(tan)
664
665            # Write face inidices
666            log(f"- LOD {lod_index + 1}: writing {face_count} faces", indent=3)
667            for i in range(0, len(face_indices), 3):
668                if i + 2 < len(face_indices):
669                    for j in range(3):
670                        chunk_data += struct.pack('I', face_indices[i + j])
671                        
672            # Cleanup
673            bpy.data.meshes.remove(temp_mesh)
674
675        return chunk_data

780            print(f"      - Spot blend: {light_data.spot_blend:.3f}")
781
782            cutoff = min(math.degrees(light_data.spot_size / 2), 40.0)
783        elif light_data.type == 'SUN':
784            cutoff = light_data.angle
785        else:
786            cutoff = 180.0
787
788        chunk_data += struct.pack('f', cutoff)
789        log(f"- Cutoff angle: {cutoff:.3f} degrees", category="LIGHT", indent=3)
790
791        # Spot exponent/falloff
792        if light_data.type == 'SPOT':
793            spot_exponent = light_data.spot_blend
794        else:
795            spot_exponent = 0.0
796
797        chunk_data += struct.pack('f', spot_exponent)
798        log(f"- Spot exponent: {spot_exponent:.3f}", category="LIGHT", indent=3)
799
800        # Cast shadows flag
801        cast_shadows = 1 if light_data.use_shadow else 0
802        chunk_data += struct.pack('B', cast_shadows)
803        log(f"- Cast shadows: {'Yes' if cast_shadows else 'No'}", category="LIGHT", indent=3)
804
805        # Volumetric flag
806        volumetric = 0
807        chunk_data += struct.pack('B', volumetric)
808        log(f"- Volumetric: {'Yes' if volumetric else 'No'}", category="LIGHT", indent=3)
809
810        # Write the chunk
811        log("- Writing light chunk to file", category="LIGHT", indent=3)
812        self.packer.write_chunk_header(file, ChunkType.LIGHT, len(chunk_data))
813        file.write(chunk_data)

 60    def convert_openGl(self, matrix):
 61
 62        matrix_copy = matrix.copy()
 63        C = mathutils.Matrix(((1,0,0,0),
 64                      (0,0,1,0),
 65                      (0,-1,0,0),
 66                      (0,0,0,1)))
 67        C_inv = C.transposed() 
 68
 69        return C @ matrix_copy @ C_inv

 86    def write_node_recursive(self, file, obj):
 87        """
 88        Writes a node and its children recursively into the OVO file.
 89        """
 90        if obj in self.processed_objects:
 91            return
 92
 93        self.processed_objects.add(obj)
 94
 95        # Process children
 96        valid_children = []
 97        for child in obj.children:
 98            if child not in self.processed_objects:
 99                if ((child.type == 'MESH' and self.use_mesh) or
100                        (child.type == 'LIGHT' and self.use_light) or
101                        (child.type not in {'MESH', 'LIGHT'})):
102                    valid_children.append(child)

105
106        if obj.type == 'MESH':
107            category = "MESH"
108        elif obj.type == 'LIGHT':
109            category = "LIGHT"
110        else:
111            category = "MESH"

115        log(f"- Children: {num_children}", category=category, indent=2)
116        log(f"- Should export: {self.should_export_object(obj)}", category=category, indent=2)
117
118        # Process materials
119        if obj.type == 'MESH' and self.should_export_object(obj):
120            for material_slot in obj.material_slots:
121                material = material_slot.material
122                if material and material not in self.processed_objects:
123                    print(f"      - Writing material: {material.name}")
124                    self.write_material_chunk(file, material)
125                    self.processed_objects.add(material)
126
127        if ((obj.type == 'MESH' and self.use_mesh) or
128                (obj.type == 'LIGHT' and self.use_light) or

130            if obj.type == 'MESH':

326    def write_mesh_chunk(self, file, obj, num_children):
327        """
328        Writes a mesh chunk to the OVO file.
329
330        Args:
331            file: Output file object
332            obj: Blender mesh object to export
333            num_children: Number of children for this node
334        """
335        chunk_data = b''
336
337        # Mesh name
338        log(f"[OVOExporter.write_mesh_chunk] Processing mesh: '{obj.name}'", category="MESH", indent=2)
339        chunk_data += self.packer.pack_string(obj.name)
340
341        # In write_mesh_chunk
342        if obj.parent:
343            local_bl = obj.matrix_local  
344        else:
345            local_bl = obj.matrix_world
346
347        # Save matrix without additional conversions
348        final_matrix = self.convert_openGl(local_bl)
349        chunk_data += self.packer.pack_matrix(final_matrix)
350
351        # Children and material data
352        chunk_data += struct.pack('I', num_children)
353        chunk_data += self.packer.pack_string("[none]")
354        chunk_data += struct.pack('B', 0)
355
356                
357        # Material assignment
358        if obj.material_slots and obj.material_slots[0].material:
359            material_name = obj.material_slots[0].material.name
360            chunk_data += self.packer.pack_string(material_name)
361            log(f"- Material: '{material_name}'", category="MESH", indent=3)
362        else:
363            chunk_data += self.packer.pack_string("[none]")
364            log("- No material assigned", category="MESH", indent=3)
365
366        # Get mesh data from evaluated object
367        log("- Getting mesh data", category="MESH", indent=3)
368        depsgraph = bpy.context.evaluated_depsgraph_get()
369        obj_eval = obj.evaluated_get(depsgraph)
370        mesh = obj_eval.to_mesh()
371
372        # Tangent
373        try:
374            mesh.calc_tangents()
375            loop_tangent = [l.tangent.copy()  for l in mesh.loops]
376            loop_sign    = [l.bitangent_sign  for l in mesh.loops]
377        except RuntimeError:
378            log("- Tangent calculation failed (n-gons detected), using safe method", category="MESH", indent=3)
379            loop_tangent, loop_sign = self.mesh_manager.safe_calc_tangents(mesh)
380
381        loop_tangent = [l.tangent.copy()      for l in mesh.loops]
382        loop_sign    = [l.bitangent_sign      for l in mesh.loops]
383        poly_lstart  = [p.loop_start          for p in mesh.polygons]
384
385        # Create two BMesh instances: original and one to be triangulated
386        import bmesh
387        original_bm = bmesh.new()
388        original_bm.from_mesh(mesh)
389
390        triangulated_bm = bmesh.new()
391        triangulated_bm.from_mesh(mesh)
392        bmesh.ops.triangulate(triangulated_bm, faces=triangulated_bm.faces)
393
394        # Ensure lookup table are updated
395        original_bm.faces.ensure_lookup_table()
396        original_bm.verts.ensure_lookup_table()
397        triangulated_bm.faces.ensure_lookup_table()
398        triangulated_bm.verts.ensure_lookup_table()
399
400        # Get UV layer
401        uv_layer = original_bm.loops.layers.uv.active
402        if uv_layer:
403            log(f"- UV layer found: '{mesh.uv_layers.active.name if mesh.uv_layers.active else 'default'}'", category="MESH", indent=3)
404        else:
405            log("- WARNING: No UV layer found", category="MESH", indent=3)
406
407        radius, min_box, max_box = self.mesh_manager.get_box_radius(mesh.verices)
408
409        # Write bounding box information
410        chunk_data += struct.pack('f', radius)
411        chunk_data += self.packer.pack_vector3(min_box)
412        chunk_data += self.packer.pack_vector3(max_box)
413
414        # Process physics data
415        log("- Processing physics data", category="MESH", indent=3)
416        chunk_data = self.physics_manager.write_physics_data(obj, chunk_data)
417        lod_manager = OVOLodManager()
418
419        # Check face count to determine if we need multi-LOD
420        should_multi_lod = lod_manager.should_generate_multi_lod(obj)
421
422        # Vertex data
423        vertex_count = 0
424        face_count = 0
425        vertices_data = []
426        face_indices = []
427
428        if should_multi_lod:
429            # Generate LOD meshes
430            log("- Generating multiple LODs for high-poly mesh", category="MESH", indent=3)
431            lod_meshes = lod_manager.generate_lod_meshes(obj)
432
433            # Write LOD data
434            chunk_data = self.write_lod_data(obj, chunk_data, lod_meshes)
435
436            # Clean up LOD meshes
437            lod_manager.cleanup_lod_meshes(lod_meshes)
438        else:
439
440            # Original single LOD code path single LOD
441            log("- LOD count: 1 (single LOD)", category="MESH", indent=3)
442            chunk_data += struct.pack('I', 1)
443
444            vertices_data = []
445            face_indices = []
446            face_vertex_map = {}
447
448            for face_idx, face in enumerate(original_bm.faces):
449                # Calculate face normals (not interpolated)
450                face_normal = face.normal.normalized()
451                
452                
453                # Iterate on faces to extract vertices data
454                for loop_idx, loop in enumerate(face.loops):
455                    vert_idx = loop.vert.index
456                    pos      = loop.vert.co
457                    uv       = loop[uv_layer].uv if uv_layer else mathutils.Vector((0.0, 0.0))
458
459                    mesh_loop_index = poly_lstart[face_idx] + loop_idx
460                    bl_tangent      = loop_tangent[mesh_loop_index]
461                    sign            = loop_sign   [mesh_loop_index]
462
463
464                    # Transform coordinates
465                    transformed_pos  = mathutils.Vector((pos.x,         pos.z,        -pos.y))
466                    transformed_norm = mathutils.Vector((face_normal.x, face_normal.z, -face_normal.y)).normalized()
467                    transformed_tan  = mathutils.Vector((bl_tangent.x,  bl_tangent.z, -bl_tangent.y)).normalized()
468
469                    # Add to lists
470                    vertex_idx = len(vertices_data)
471                    vertices_data.append((transformed_pos, transformed_norm, uv,
472                                        transformed_tan, sign))
473
474                    face_vertex_map[(face_idx, vert_idx)] = vertex_idx
475
476            
477            # Indices for triangulated faces 
478            face_indices = []
479
480            log("- Manual triangulate started", category="MESH", indent=3)
481            face_count = 0
482            for face_idx, face in enumerate(original_bm.faces):
483                # If we have a quad (4 vertices)
484                if len(face.verts) == 4:
485                
486                    v_indices = [face_vertex_map.get((face_idx, v.index), 0) for v in face.verts]
487                    face_indices.extend([v_indices[0], v_indices[1], v_indices[2]])
488                    face_indices.extend([v_indices[0], v_indices[2], v_indices[3]])
489                    face_count += 2
490
491                elif len(face.verts) == 3:
492                    # If we already have a tris 
493                    v_indices = [face_vertex_map.get((face_idx, v.index), 0) for v in face.verts]
494                    face_indices.extend(v_indices)
495                    face_count += 1
496                else:
497                    # For n-gons faces
498                    v0_idx = face_vertex_map.get((face_idx, face.verts[0].index), 0)
499                    for i in range(1, len(face.verts) - 1):
500                        v1_idx = face_vertex_map.get((face_idx, face.verts[i].index), 0)
501                        v2_idx = face_vertex_map.get((face_idx, face.verts[i+1].index), 0)
502                        face_indices.extend([v0_idx, v1_idx, v2_idx])
503                        face_count += 1
504
505            # write faces and vertices
506            vertex_count = len(vertices_data)
507            log(f"- Vertices: {vertex_count}", category="MESH", indent=3)
508            log(f"- Triangulated faces: {face_count}", category="MESH", indent=3)
509            chunk_data += struct.pack('I', vertex_count)
510            chunk_data += struct.pack('I', face_count)
511
512        # If we have a single lod
513        if not should_multi_lod:
514            log(f"- Writing {vertex_count} vertices", category="MESH", indent=3)
515        
516            # Write vertex data
517            for pos, norm, uv, tan, sign in vertices_data:
518                chunk_data += self.packer.pack_vector3(pos)
519                chunk_data += self.packer.pack_normal(norm)
520                chunk_data += self.packer.pack_uv(uv)
521                chunk_data += self.packer.pack_tangent(tan)
522
523            # Write face data
524            log(f" - Writing {face_count} faces", category="MESH", indent=3)
525            for i in range(0, len(face_indices), 3):
526                if i + 2 < len(face_indices):
527                    for j in range(3):
528                        chunk_data += struct.pack('I', face_indices[i + j])
529
530        # Write the complete mesh chunk
531        log(f"- Writing mesh chunk to file", category="MESH", indent=3)
532        self.packer.write_chunk_header(file, ChunkType.MESH, len(chunk_data))
533        file.write(chunk_data)
534
535        # Cleanup
536        original_bm.free()
537        triangulated_bm.free()
538        obj_eval.to_mesh_clear()
539        log(f"[OVOExporter.write_mesh_chunk] Completed: '{obj.name}'", indent=3)

544    def write_lod_data(self, obj, chunk_data, lod_meshes):
545        """
546        Writes LOD data to the chunk.
547
548        Args:
549            obj: Blender mesh object
550            chunk_data: Current chunk data buffer
551            lod_meshes: List of BMesh objects for each LOD
552
553        Returns:
554            bytes: Updated chunk data with LOD information
555        """
556        # Write the number of LODs
557        lod_count = len(lod_meshes)
558        chunk_data += struct.pack('I', lod_count)
559        log(f"- LOD count: {lod_count}", indent=3)
560
561        # Process each LOD level
562        for lod_index, bm in enumerate(lod_meshes):
563            log(f"- Processing LOD {lod_index + 1}/{lod_count}", indent=3)
564
565            # Convert original mesh to BMesh to calculate tangents
566            temp_mesh = bpy.data.meshes.new(f"temp_lod_{lod_index}")
567            bm.to_mesh(temp_mesh)
568
569            # Tangents
570            try:
571                temp_mesh.calc_tangents()
572                loop_tangent = [l.tangent.copy()  for l in temp_mesh.loops]
573                loop_sign    = [l.bitangent_sign  for l in temp_mesh.loops]
574            except RuntimeError:
575                log(f"- LOD {lod_index + 1}: mesh.calc_tangents() failes: using secure fallback method", indent=3)
576                loop_tangent, loop_sign = self.mesh_manager.safe_calc_tangents(temp_mesh)
577            
578            # Get loop start index for each polygon
579            poly_lstart = [p.loop_start for p in temp_mesh.polygons]
580
581            # Get the UV layer
582            uv_layer = bm.loops.layers.uv.active
583            if uv_layer:
584                log(f"- LOD {lod_index + 1}: UV layer found: '{temp_mesh.uv_layers.active.name if temp_mesh.uv_layers.active else 'default'}'", indent=3)
585            else:
586                log(f"- LOD {lod_index + 1}: WARNING - No UV layer found", indent=3)
587
588            # Vertex data
589            vertices_data = []
590            face_indices = []
591            face_vertex_map = {}
592
593            # Process faces
594            log(f"- LOD {lod_index + 1}: Processando le facce originali", indent=3)
595
596            for face_idx, face in enumerate(bm.faces):
597                
598                face_normal = face.normal.normalized()
599                
600                # For each vertex in this face
601                for loop_idx, loop in enumerate(face.loops):
602                    vert_idx = loop.vert.index
603                    pos = loop.vert.co
604                    uv = loop[uv_layer].uv if uv_layer else mathutils.Vector((0.0, 0.0))
605
606                    # Get loop face index
607                    mesh_loop_index = poly_lstart[face_idx] + loop_idx
608                    if mesh_loop_index < len(loop_tangent):
609                        bl_tangent = loop_tangent[mesh_loop_index]
610                        sign = loop_sign[mesh_loop_index]
611                    else:
612                        # fallback
613                        bl_tangent = mathutils.Vector((1.0, 0.0, 0.0))
614                        sign = 1.0
615
616                    # Transform coordinates for OpenGL
617                    transformed_pos = mathutils.Vector((pos.x, pos.z, -pos.y))
618                    transformed_norm = mathutils.Vector((face_normal.x, face_normal.z, -face_normal.y)).normalized()
619                    transformed_tan = mathutils.Vector((bl_tangent.x, bl_tangent.z, -bl_tangent.y)).normalized()
620
621                    # Create a unique vertex
622                    vertex_idx = len(vertices_data)
623                    vertices_data.append((transformed_pos, transformed_norm, uv, transformed_tan, sign))
624                    face_vertex_map[(face_idx, vert_idx)] = vertex_idx
625
626            # Triangulate faces
627            log(f"- LOD triangulating faces", indent=3)
628            face_count = 0
629            for face_idx, face in enumerate(bm.faces):
630                # If we have a quad (4 vertices)
631                if len(face.verts) == 4:
632                    
633                    v_indices = [face_vertex_map.get((face_idx, v.index), 0) for v in face.verts]
634                    
635                    face_indices.extend([v_indices[0], v_indices[1], v_indices[2]])
636                    face_indices.extend([v_indices[0], v_indices[2], v_indices[3]])
637                    face_count += 2 
638                elif len(face.verts) == 3:
639                    # If we already have a tris
640                    v_indices = [face_vertex_map.get((face_idx, v.index), 0) for v in face.verts]
641                    face_indices.extend(v_indices)
642                    face_count += 1 
643                else:
644                    # For n-gons faces
645                    v0_idx = face_vertex_map.get((face_idx, face.verts[0].index), 0)
646                    for i in range(1, len(face.verts) - 1):
647                        v1_idx = face_vertex_map.get((face_idx, face.verts[i].index), 0)
648                        v2_idx = face_vertex_map.get((face_idx, face.verts[i+1].index), 0)
649                        face_indices.extend([v0_idx, v1_idx, v2_idx])
650                        face_count += 1 
651
652            # Write vertices and faces count
653            vertex_count = len(vertices_data)
654            log(f"- LOD {lod_index + 1}: {vertex_count} vertices, {face_count} faces", indent=3)
655            chunk_data += struct.pack('I', vertex_count)
656            chunk_data += struct.pack('I', face_count)
657
658            # Write vertex data
659            log(f"- LOD {lod_index + 1}: Writing {vertex_count} vertices", indent=3)
660            for pos, norm, uv, tan, sign in vertices_data:
661                chunk_data += self.packer.pack_vector3(pos)
662                chunk_data += self.packer.pack_normal(norm)
663                chunk_data += self.packer.pack_uv(uv)
664                chunk_data += self.packer.pack_tangent(tan)
665
666            # Write face inidices
667            log(f"- LOD {lod_index + 1}: writing {face_count} faces", indent=3)
668            for i in range(0, len(face_indices), 3):
669                if i + 2 < len(face_indices):
670                    for j in range(3):
671                        chunk_data += struct.pack('I', face_indices[i + j])
672                        
673            # Cleanup
674            bpy.data.meshes.remove(temp_mesh)
675
676        return chunk_data

779            print(f"      - Spot size (radians): {light_data.spot_size:.3f}")

781            print(f"      - Spot blend: {light_data.spot_blend:.3f}")
782

788

791

794            spot_exponent = light_data.spot_blend
795        else:
796            spot_exponent = 0.0
797
798        chunk_data += struct.pack('f', spot_exponent)
799        log(f"- Spot exponent: {spot_exponent:.3f}", category="LIGHT", indent=3)
800
801        # Cast shadows flag
802        cast_shadows = 1 if light_data.use_shadow else 0
803        chunk_data += struct.pack('B', cast_shadows)
804        log(f"- Cast shadows: {'Yes' if cast_shadows else 'No'}", category="LIGHT", indent=3)
805

808        chunk_data += struct.pack('B', volumetric)
809        log(f"- Volumetric: {'Yes' if volumetric else 'No'}", category="LIGHT", indent=3)
810
811        # Write the chunk
812        log("- Writing light chunk to file", category="LIGHT", indent=3)
813        self.packer.write_chunk_header(file, ChunkType.LIGHT, len(chunk_data))
