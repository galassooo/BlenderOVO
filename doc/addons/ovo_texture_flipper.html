123    # Flip BC1 data block
124    # --------------------------------------------------------
125    @staticmethod
126    def _flip_bc1_block(block: bytes) -> bytes:
127        """Vertically flips a BC1 / DXT1 block (8 bytes)."""
128        # byte 0-3 = color0 & color1 (remain unchanged)
129        colour_hdr = block[:4]
130        # byte 4-7 = 16 2-bit color indices (row-major, little-endian)
131        idx = int.from_bytes(block[4:], 'little')
132        # extract 4 rows of 8 bits each and flip their order
133        r0 = (idx >>  0) & 0xFF
134        r1 = (idx >>  8) & 0xFF
135        r2 = (idx >> 16) & 0xFF
136        r3 = (idx >> 24) & 0xFF
137        flipped = (r3 | (r2 << 8) | (r1 << 16) | (r0 << 24))
138        return colour_hdr + flipped.to_bytes(4, 'little')

141    # --------------------------------------------------------
142    # Flip BC3 data block (with alpha)
143    # --------------------------------------------------------
144    @staticmethod
145    def _flip_bc3_alpha(block: bytes) -> bytes:
146        """Vertically flips the 6 bytes of alpha indices used in DXT5 / BC3 / BC4 / BC5."""
147        a_idx = int.from_bytes(block, 'little')  # 48-bit alpha index
148        rows = [(a_idx >> (12 * i)) & 0xFFF for i in range(4)]
149        flipped = sum(rows[i] << (12 * (3 - i)) for i in range(4))
150        return flipped.to_bytes(6, 'little')
151
152    # --------------------------------------------------------
153    # Flip BC3 data block (without alpha)
154    # --------------------------------------------------------
155    @staticmethod
156    def _flip_bc3_block(block: bytes) -> bytes:
157        """Vertically flips a BC3 / DXT5 block (16 bytes)."""
158        # 0-1 = alpha0/alpha1, 2-7 = alpha indices (48 bits)
159        a0a1 = block[0:2]
160        a_idx = OVOTextureFlipper._flip_bc3_alpha(block[2:8])
161        # 8-15 = color block (same structure as BC1)
162        colour = OVOTextureFlipper._flip_bc1_block(block[8:])
163        return a0a1 + a_idx + colour

165    # --------------------------------------------------------
166    # Flip BC5 data block (with alpha)
167    # --------------------------------------------------------
168    @staticmethod
169    def _flip_bc5_block(block: bytes) -> bytes:
170        """Vertically flips a BC5 block (16 bytes = R-channel + G-channel)."""
171        # 0-7   = R channel (similar to alpha block in BC3)
172        # 8-15  = G channel (same format)
173        r_part = block[:8]
174        g_part = block[8:]
175        r_flipped = r_part[:2] + OVOTextureFlipper._flip_bc3_alpha(r_part[2:])
176        g_flipped = g_part[:2] + OVOTextureFlipper._flip_bc3_alpha(g_part[2:])
177        return r_flipped + g_flipped

180    # Flip DDS Texture
181    # --------------------------------------------------------
182    @staticmethod
183    def flip_dds_texture(input_path, output_path=None):
184        """
185        Vertically flips a DDS texture, supporting standard formats and DX10 headers.
186
187        Args:
188            input_path (str): Path to the DDS file to flip.
189            output_path (str, optional): Output path. If None, overwrites the original file.
190
191        Returns:
192            str: Path to the flipped DDS file, or original file on error.
193        """
194                
195        if not os.path.exists(input_path):
196            log(f"File does not exist: {input_path}", category="ERROR")
197            return input_path
198
199
200        if output_path is None: #if no output is specified, overwrite the input file
201            temp_dir = tempfile.gettempdir()  #create a copy
202            temp_file = os.path.join(temp_dir, f"temp_{os.path.basename(input_path)}")
203            final_output = input_path
204        else:
205            temp_file = output_path
206            final_output = output_path
207
208        try:
209
210            # Read file
211            with open(input_path, 'rb') as f:
212                data = f.read()
213
214            # Verify input file is DDS
215            if len(data) < OVOTextureFlipper.HEADER_SIZE or data[:4] != b'DDS ':
216                raise ValueError(f"Il file {input_path} non  un file DDS valido")
217
218            # Extract header information
219            header = data[:OVOTextureFlipper.HEADER_SIZE]
220            width = struct.unpack('<I', data[OVOTextureFlipper.WIDTH_OFFSET:OVOTextureFlipper.WIDTH_OFFSET+4])[0]
221            height = struct.unpack('<I', data[OVOTextureFlipper.HEIGHT_OFFSET:OVOTextureFlipper.HEIGHT_OFFSET+4])[0]
222            flags = struct.unpack('<I', data[OVOTextureFlipper.FLAGS_OFFSET:OVOTextureFlipper.FLAGS_OFFSET+4])[0]
223            has_mipmap = (flags & OVOTextureFlipper.DDSD_MIPMAPCOUNT) != 0
224            mipmap_count = struct.unpack('<I', data[OVOTextureFlipper.MIPMAP_COUNT_OFFSET:OVOTextureFlipper.MIPMAP_COUNT_OFFSET+4])[0] if has_mipmap else 1
225
226            # Extract pixel infos
227            pixel_format_flags = struct.unpack('<I', data[OVOTextureFlipper.PIXEL_FORMAT_OFFSET:OVOTextureFlipper.PIXEL_FORMAT_OFFSET+4])[0]
228            four_cc = data[OVOTextureFlipper.FOURCC_OFFSET:OVOTextureFlipper.FOURCC_OFFSET+4]
229
230            log(f"Flipping DDS: {os.path.basename(input_path)}", category="")
231            log(f"  Size: {width}x{height}", category="", indent=1)
232            log(f"  Mipmaps: {mipmap_count}", category="", indent=1)
233            log(f"  FourCC: {four_cc.decode('ascii', errors='replace')}", category="", indent=1)
234
235            # Determine if the DDS is a DX10 
236            header_size = OVOTextureFlipper.HEADER_SIZE
237            dxgi_format = None
238            is_dx10 = False
239
240            # Check for DX10 header
241            if four_cc == OVOTextureFlipper.DX10_FOURCC:
242                is_dx10 = True
243                header_size += OVOTextureFlipper.DX10_HEADER_SIZE 
244
245                # Extract DXGI_FORMAT from header
246                dxgi_format = struct.unpack('<I', data[OVOTextureFlipper.HEADER_SIZE:OVOTextureFlipper.HEADER_SIZE+4])[0]
247                log(f"  DX10 header detected. DXGI_FORMAT: {dxgi_format} ({OVOTextureFlipper.DXGI_FORMAT.get(dxgi_format, 'Unknown')})",category="", indent=1)
248
249            # Determine block size basing on the type
250            block_size = None
251
252            if is_dx10:
253                # If DX10, use DXGI format to determine size
254                if dxgi_format in OVOTextureFlipper.DXGI_BLOCK_SIZE:
255                    block_size = OVOTextureFlipper.DXGI_BLOCK_SIZE[dxgi_format]
256                    log(f"  Using block size: {block_size} bytes (DXGI_FORMAT: {OVOTextureFlipper.DXGI_FORMAT.get(dxgi_format)})",category="", indent=1)
257                else:
258                    raise ValueError(f"DXGI_FORMAT non supportato: {dxgi_format}")
259            else:
260                # Use Four CC for other formats
261                if four_cc in OVOTextureFlipper.BLOCK_SIZE:
262                    block_size = OVOTextureFlipper.BLOCK_SIZE[four_cc]
263                    log(f"  Using block size: {block_size} bytes for format: {four_cc.decode('ascii', errors='replace')}",category="", indent=1)
264                else:
265                    # For uncompressed formats or other formats, try to determine the size basing on pitch
266                    if flags & OVOTextureFlipper.DDSD_PITCH:
267                        raise ValueError(f"Formato non compresso con pitch non supportato")
268                    elif flags & OVOTextureFlipper.DDSD_LINEARSIZE:
269                        # Try to compute size by linear 
270                        linear_size = struct.unpack('<I', data[OVOTextureFlipper.PITCH_OR_LINEAR_SIZE_OFFSET:OVOTextureFlipper.PITCH_OR_LINEAR_SIZE_OFFSET+4])[0]
271                        width_blocks = (width + 3) // 4
272                        height_blocks = (height + 3) // 4
273                        total_blocks = width_blocks * height_blocks
274                        if total_blocks > 0:
275                            estimated_block_size = linear_size / total_blocks
276                            if estimated_block_size <= 12:
277                                block_size = 8
278                            else:
279                                block_size = 16
280                            log(f"  Estimated block size: {estimated_block_size:.2f}  rounded to {block_size}", category="WARNING", indent=1)
281                        else:
282                            raise ValueError(f"Impossibile determinare la dimensione del blocco")
283                    else:
284                        raise ValueError(f"Formato non supportato: {four_cc}")
285
286            # Write new file
287            with open(temp_file, 'wb') as f:
288            
289                # Write header
290                f.write(data[:header_size])
291
292                pos = header_size
293
294                # For each mipmap
295                current_width, current_height = width, height
296                for level in range(mipmap_count):
297
298                    # Block size
299                    width_blocks = max(1, (current_width + 3) // 4)
300                    height_blocks = max(1, (current_height + 3) // 4)
301
302                    # Level dimension
303                    mipmap_size = width_blocks * height_blocks * block_size
304
305                    # Ensure all data is available
306                    if pos + mipmap_size > len(data):
307                        log(f"  WARNING: Mipmap {level} data appears incomplete", category="WARNING", indent=1)
308                        # copy data and exit
309                        f.write(data[pos:])
310                        break
311
312                    # Extract data from mipmap
313                    mipmap_data = data[pos:pos+mipmap_size]
314
315                    # Calculate row length in block
316                    row_size = width_blocks * block_size
317
318                    # Divides data in rows
319                    flipped_rows = []
320                    for row in range(height_blocks): #From top to bottom
321                        start = row * row_size
322                        end   = start + row_size
323                        row_data = mipmap_data[start:end]
324                        new_row  = bytearray()
325
326                        # Iterate through all row blocks
327                        for col in range(width_blocks):
328                            b_start = col * block_size
329                            b_end   = b_start + block_size
330                            blk = row_data[b_start:b_end]
331
332                            # Flip block pixels depending on block size and type
333                            if block_size == 8:
334                                blk = OVOTextureFlipper._flip_bc1_block(blk)
335                            else:  # 16 byte
336                                if four_cc in (OVOTextureFlipper.DXT5_FOURCC,) or \
337                                   dxgi_format in (77, 78, 79):
338                                    blk = OVOTextureFlipper._flip_bc3_block(blk)
339                                elif four_cc in (OVOTextureFlipper.BC5_FOURCC,
340                                                  OVOTextureFlipper.BC5U_FOURCC,
341                                                  OVOTextureFlipper.BC5S_FOURCC) or \
342                                     dxgi_format in (83, 84, 85):
343                                    blk = OVOTextureFlipper._flip_bc5_block(blk)
344                                
345                            new_row.extend(blk)
346
347                        flipped_rows.append(bytes(new_row))
348
349                    # Flip block orders
350                    flipped_mipmap = b''.join(reversed(flipped_rows))
351
352                    # Write flipped mipmap
353                    f.write(flipped_mipmap)
354
355                    # Detailed log only for first level:
356                    if level == 0:
357                        log(f"  Base mipmap: {width_blocks}x{height_blocks} blocks ({row_size} bytes/row)",category="", indent=1)
358
359                    # Update cursor position to swith to the next mipmap level
360                    pos += mipmap_size
361
362                    # Divide by 2 dimension (mipmaps size are always / 2, eg. first original, second original/2, third original/4...)
363                    current_width = max(1, current_width // 2)
364                    current_height = max(1, current_height // 2)
365
366                # If there are additional data copy them
367                if pos < len(data):
368                    remaining_data = data[pos:]
369                    f.write(remaining_data)
370                    log(f"  Copied {len(remaining_data)} extra bytes after mipmaps", category="", indent=1)
371
372            # If we need to overwrite the orignal file, copy the flipped image over
373            if temp_file != final_output:
374                shutil.move(temp_file, final_output)
375
376            log(f"Texture successfully flipped: {final_output}", category="")
377            return final_output
378
379        except (IOError, ValueError) as e:
380            log(f"ERROR during texture flipping: {str(e)}", category="ERROR")
381
382            if temp_file != input_path and output_path is None:
383                if os.path.exists(temp_file):
384                    try:
385                        os.remove(temp_file)
386                    except:
387                        pass
388            return input_path
389        except Exception as e:
390            log(f"UNEXPECTED ERROR during texture flip: {str(e)}", category="ERROR")
391
392            return input_path

417    # Get DDS Texture Info
418    # --------------------------------------------------------
419    @staticmethod
420    def get_dds_info(filepath):
421        """
422        Extracts basic information from a DDS file.
423
424        Args:
425            filepath (str): Path to the DDS file.
426
427        Returns:
428            dict: Dictionary with info (width, height, mipmap_count, format, dxgi_format if present),
429                  or None if the file is not a valid DDS.
430        """
431        try:
432            with open(filepath, 'rb') as f:
433                data = f.read(148)
434
435            if data[:4] != b'DDS ':
436                return None
437
438            width = struct.unpack('<I', data[OVOTextureFlipper.WIDTH_OFFSET:OVOTextureFlipper.WIDTH_OFFSET+4])[0]
439            height = struct.unpack('<I', data[OVOTextureFlipper.HEIGHT_OFFSET:OVOTextureFlipper.HEIGHT_OFFSET+4])[0]
440            flags = struct.unpack('<I', data[OVOTextureFlipper.FLAGS_OFFSET:OVOTextureFlipper.FLAGS_OFFSET+4])[0]
441            has_mipmap = (flags & OVOTextureFlipper.DDSD_MIPMAPCOUNT) != 0
442            mipmap_count = struct.unpack('<I', data[OVOTextureFlipper.MIPMAP_COUNT_OFFSET:OVOTextureFlipper.MIPMAP_COUNT_OFFSET+4])[0] if has_mipmap else 1
443            four_cc = data[OVOTextureFlipper.FOURCC_OFFSET:OVOTextureFlipper.FOURCC_OFFSET+4]
444
445            result = {
446                'width': width,
447                'height': height,
448                'mipmap_count': mipmap_count,
449                'format': four_cc
450            }
451
452            # If it is a DX10, extract the DXGI_FORMAT
453            if four_cc == OVOTextureFlipper.DX10_FOURCC and len(data) >= OVOTextureFlipper.HEADER_SIZE + 4:
454                dxgi_format = struct.unpack('<I', data[OVOTextureFlipper.HEADER_SIZE:OVOTextureFlipper.HEADER_SIZE+4])[0]
455                result['dxgi_format'] = dxgi_format
456                result['dxgi_format_name'] = OVOTextureFlipper.DXGI_FORMAT.get(dxgi_format, 'Unknown')
457
458            return result
459        except Exception as e:
460            log(f"Error analyzing DDS file: {str(e)}", category="ERROR")
461            return None

466    @staticmethod
467    def safe_flip_dds_texture(input_path, output_path=None):
468        """
469        A safe wrapper for flip_dds_texture that does not raise exceptions.
470
471        Args:
472            input_path (str): Path to the DDS texture to flip.
473            output_path (str, optional): Output path. If None, will overwrite the original.
474
475        Returns:
476            tuple: (bool, str)  Success flag and resulting path.
477        """
478        if not input_path or not os.path.exists(input_path):
479            return False, input_path
480
481        try:
482            # Flip the texture
483            result_path = OVOTextureFlipper.flip_dds_texture(input_path, output_path)
484
485            # Verify if the resulting file exists
486            if result_path != input_path and os.path.exists(result_path):
487                return True, result_path
488            elif os.path.exists(input_path):
489                return False, input_path
490            else:
491                return False, None
492        except Exception as e:
493            log(f"Exception in safe texture flip: {str(e)}", category="ERROR")
494            return False, input_path

123    # --------------------------------------------------------
124    # Flip BC1 data block
125    # --------------------------------------------------------
126    @staticmethod
127    def _flip_bc1_block(block: bytes) -> bytes:
128        """Vertically flips a BC1 / DXT1 block (8 bytes)."""
129        # byte 0-3 = color0 & color1 (remain unchanged)
130        colour_hdr = block[:4]
131        # byte 4-7 = 16 2-bit color indices (row-major, little-endian)
132        idx = int.from_bytes(block[4:], 'little')
133        # extract 4 rows of 8 bits each and flip their order
134        r0 = (idx >>  0) & 0xFF
135        r1 = (idx >>  8) & 0xFF
136        r2 = (idx >> 16) & 0xFF
137        r3 = (idx >> 24) & 0xFF
138        flipped = (r3 | (r2 << 8) | (r1 << 16) | (r0 << 24))
139        return colour_hdr + flipped.to_bytes(4, 'little')

142    # --------------------------------------------------------
143    # Flip BC3 data block (with alpha)
144    # --------------------------------------------------------
145    @staticmethod
146    def _flip_bc3_alpha(block: bytes) -> bytes:
147        """Vertically flips the 6 bytes of alpha indices used in DXT5 / BC3 / BC4 / BC5."""
148        a_idx = int.from_bytes(block, 'little')  # 48-bit alpha index
149        rows = [(a_idx >> (12 * i)) & 0xFFF for i in range(4)]
150        flipped = sum(rows[i] << (12 * (3 - i)) for i in range(4))
151        return flipped.to_bytes(6, 'little')
152
153    # --------------------------------------------------------
154    # Flip BC3 data block (without alpha)
155    # --------------------------------------------------------
156    @staticmethod
157    def _flip_bc3_block(block: bytes) -> bytes:
158        """Vertically flips a BC3 / DXT5 block (16 bytes)."""
159        # 0-1 = alpha0/alpha1, 2-7 = alpha indices (48 bits)
160        a0a1 = block[0:2]
161        a_idx = OVOTextureFlipper._flip_bc3_alpha(block[2:8])
162        # 8-15 = color block (same structure as BC1)
163        colour = OVOTextureFlipper._flip_bc1_block(block[8:])
164        return a0a1 + a_idx + colour

166    # --------------------------------------------------------
167    # Flip BC5 data block (with alpha)
168    # --------------------------------------------------------
169    @staticmethod
170    def _flip_bc5_block(block: bytes) -> bytes:
171        """Vertically flips a BC5 block (16 bytes = R-channel + G-channel)."""
172        # 0-7   = R channel (similar to alpha block in BC3)
173        # 8-15  = G channel (same format)
174        r_part = block[:8]
175        g_part = block[8:]
176        r_flipped = r_part[:2] + OVOTextureFlipper._flip_bc3_alpha(r_part[2:])
177        g_flipped = g_part[:2] + OVOTextureFlipper._flip_bc3_alpha(g_part[2:])
178        return r_flipped + g_flipped

183    @staticmethod
184    def flip_dds_texture(input_path, output_path=None):
185        """
186        Vertically flips a DDS texture, supporting standard formats and DX10 headers.
187
188        Args:
189            input_path (str): Path to the DDS file to flip.
190            output_path (str, optional): Output path. If None, overwrites the original file.
191
192        Returns:
193            str: Path to the flipped DDS file, or original file on error.
194        """
195                
196        if not os.path.exists(input_path):
197            log(f"File does not exist: {input_path}", category="ERROR")
198            return input_path
199
200
201        if output_path is None: #if no output is specified, overwrite the input file
202            temp_dir = tempfile.gettempdir()  #create a copy
203            temp_file = os.path.join(temp_dir, f"temp_{os.path.basename(input_path)}")
204            final_output = input_path
205        else:
206            temp_file = output_path
207            final_output = output_path
208
209        try:
210
211            # Read file
212            with open(input_path, 'rb') as f:
213                data = f.read()
214
215            # Verify input file is DDS
216            if len(data) < OVOTextureFlipper.HEADER_SIZE or data[:4] != b'DDS ':
217                raise ValueError(f"Il file {input_path} non  un file DDS valido")
218
219            # Extract header information
220            header = data[:OVOTextureFlipper.HEADER_SIZE]
221            width = struct.unpack('<I', data[OVOTextureFlipper.WIDTH_OFFSET:OVOTextureFlipper.WIDTH_OFFSET+4])[0]
222            height = struct.unpack('<I', data[OVOTextureFlipper.HEIGHT_OFFSET:OVOTextureFlipper.HEIGHT_OFFSET+4])[0]
223            flags = struct.unpack('<I', data[OVOTextureFlipper.FLAGS_OFFSET:OVOTextureFlipper.FLAGS_OFFSET+4])[0]
224            has_mipmap = (flags & OVOTextureFlipper.DDSD_MIPMAPCOUNT) != 0
225            mipmap_count = struct.unpack('<I', data[OVOTextureFlipper.MIPMAP_COUNT_OFFSET:OVOTextureFlipper.MIPMAP_COUNT_OFFSET+4])[0] if has_mipmap else 1
226
227            # Extract pixel infos
228            pixel_format_flags = struct.unpack('<I', data[OVOTextureFlipper.PIXEL_FORMAT_OFFSET:OVOTextureFlipper.PIXEL_FORMAT_OFFSET+4])[0]
229            four_cc = data[OVOTextureFlipper.FOURCC_OFFSET:OVOTextureFlipper.FOURCC_OFFSET+4]
230
231            log(f"Flipping DDS: {os.path.basename(input_path)}", category="")
232            log(f"  Size: {width}x{height}", category="", indent=1)
233            log(f"  Mipmaps: {mipmap_count}", category="", indent=1)
234            log(f"  FourCC: {four_cc.decode('ascii', errors='replace')}", category="", indent=1)
235
236            # Determine if the DDS is a DX10 
237            header_size = OVOTextureFlipper.HEADER_SIZE
238            dxgi_format = None
239            is_dx10 = False
240
241            # Check for DX10 header
242            if four_cc == OVOTextureFlipper.DX10_FOURCC:
243                is_dx10 = True
244                header_size += OVOTextureFlipper.DX10_HEADER_SIZE 
245
246                # Extract DXGI_FORMAT from header
247                dxgi_format = struct.unpack('<I', data[OVOTextureFlipper.HEADER_SIZE:OVOTextureFlipper.HEADER_SIZE+4])[0]
248                log(f"  DX10 header detected. DXGI_FORMAT: {dxgi_format} ({OVOTextureFlipper.DXGI_FORMAT.get(dxgi_format, 'Unknown')})",category="", indent=1)
249
250            # Determine block size basing on the type
251            block_size = None
252
253            if is_dx10:
254                # If DX10, use DXGI format to determine size
255                if dxgi_format in OVOTextureFlipper.DXGI_BLOCK_SIZE:
256                    block_size = OVOTextureFlipper.DXGI_BLOCK_SIZE[dxgi_format]
257                    log(f"  Using block size: {block_size} bytes (DXGI_FORMAT: {OVOTextureFlipper.DXGI_FORMAT.get(dxgi_format)})",category="", indent=1)
258                else:
259                    raise ValueError(f"DXGI_FORMAT non supportato: {dxgi_format}")
260            else:
261                # Use Four CC for other formats
262                if four_cc in OVOTextureFlipper.BLOCK_SIZE:
263                    block_size = OVOTextureFlipper.BLOCK_SIZE[four_cc]
264                    log(f"  Using block size: {block_size} bytes for format: {four_cc.decode('ascii', errors='replace')}",category="", indent=1)
265                else:
266                    # For uncompressed formats or other formats, try to determine the size basing on pitch
267                    if flags & OVOTextureFlipper.DDSD_PITCH:
268                        raise ValueError(f"Formato non compresso con pitch non supportato")
269                    elif flags & OVOTextureFlipper.DDSD_LINEARSIZE:
270                        # Try to compute size by linear 
271                        linear_size = struct.unpack('<I', data[OVOTextureFlipper.PITCH_OR_LINEAR_SIZE_OFFSET:OVOTextureFlipper.PITCH_OR_LINEAR_SIZE_OFFSET+4])[0]
272                        width_blocks = (width + 3) // 4
273                        height_blocks = (height + 3) // 4
274                        total_blocks = width_blocks * height_blocks
275                        if total_blocks > 0:
276                            estimated_block_size = linear_size / total_blocks
277                            if estimated_block_size <= 12:
278                                block_size = 8
279                            else:
280                                block_size = 16
281                            log(f"  Estimated block size: {estimated_block_size:.2f}  rounded to {block_size}", category="WARNING", indent=1)
282                        else:
283                            raise ValueError(f"Impossibile determinare la dimensione del blocco")
284                    else:
285                        raise ValueError(f"Formato non supportato: {four_cc}")
286
287            # Write new file
288            with open(temp_file, 'wb') as f:
289            
290                # Write header
291                f.write(data[:header_size])
292
293                pos = header_size
294
295                # For each mipmap
296                current_width, current_height = width, height
297                for level in range(mipmap_count):
298
299                    # Block size
300                    width_blocks = max(1, (current_width + 3) // 4)
301                    height_blocks = max(1, (current_height + 3) // 4)
302
303                    # Level dimension
304                    mipmap_size = width_blocks * height_blocks * block_size
305
306                    # Ensure all data is available
307                    if pos + mipmap_size > len(data):
308                        log(f"  WARNING: Mipmap {level} data appears incomplete", category="WARNING", indent=1)
309                        # copy data and exit
310                        f.write(data[pos:])
311                        break
312
313                    # Extract data from mipmap
314                    mipmap_data = data[pos:pos+mipmap_size]
315
316                    # Calculate row length in block
317                    row_size = width_blocks * block_size
318
319                    # Divides data in rows
320                    flipped_rows = []
321                    for row in range(height_blocks): #From top to bottom
322                        start = row * row_size
323                        end   = start + row_size
324                        row_data = mipmap_data[start:end]
325                        new_row  = bytearray()
326
327                        # Iterate through all row blocks
328                        for col in range(width_blocks):
329                            b_start = col * block_size
330                            b_end   = b_start + block_size
331                            blk = row_data[b_start:b_end]
332
333                            # Flip block pixels depending on block size and type
334                            if block_size == 8:
335                                blk = OVOTextureFlipper._flip_bc1_block(blk)
336                            else:  # 16 byte
337                                if four_cc in (OVOTextureFlipper.DXT5_FOURCC,) or \
338                                   dxgi_format in (77, 78, 79):
339                                    blk = OVOTextureFlipper._flip_bc3_block(blk)
340                                elif four_cc in (OVOTextureFlipper.BC5_FOURCC,
341                                                  OVOTextureFlipper.BC5U_FOURCC,
342                                                  OVOTextureFlipper.BC5S_FOURCC) or \
343                                     dxgi_format in (83, 84, 85):
344                                    blk = OVOTextureFlipper._flip_bc5_block(blk)
345                                
346                            new_row.extend(blk)
347
348                        flipped_rows.append(bytes(new_row))
349
350                    # Flip block orders
351                    flipped_mipmap = b''.join(reversed(flipped_rows))
352
353                    # Write flipped mipmap
354                    f.write(flipped_mipmap)
355
356                    # Detailed log only for first level:
357                    if level == 0:
358                        log(f"  Base mipmap: {width_blocks}x{height_blocks} blocks ({row_size} bytes/row)",category="", indent=1)
359
360                    # Update cursor position to swith to the next mipmap level
361                    pos += mipmap_size
362
363                    # Divide by 2 dimension (mipmaps size are always / 2, eg. first original, second original/2, third original/4...)
364                    current_width = max(1, current_width // 2)
365                    current_height = max(1, current_height // 2)
366
367                # If there are additional data copy them
368                if pos < len(data):
369                    remaining_data = data[pos:]
370                    f.write(remaining_data)
371                    log(f"  Copied {len(remaining_data)} extra bytes after mipmaps", category="", indent=1)
372
373            # If we need to overwrite the orignal file, copy the flipped image over
374            if temp_file != final_output:
375                shutil.move(temp_file, final_output)
376
377            log(f"Texture successfully flipped: {final_output}", category="")
378            return final_output
379
380        except (IOError, ValueError) as e:
381            log(f"ERROR during texture flipping: {str(e)}", category="ERROR")
382
383            if temp_file != input_path and output_path is None:
384                if os.path.exists(temp_file):
385                    try:
386                        os.remove(temp_file)
387                    except:
388                        pass
389            return input_path
390        except Exception as e:
391            log(f"UNEXPECTED ERROR during texture flip: {str(e)}", category="ERROR")
392
393            return input_path

255                if dxgi_format in OVOTextureFlipper.DXGI_BLOCK_SIZE:
256                    block_size = OVOTextureFlipper.DXGI_BLOCK_SIZE[dxgi_format]
257                    log(f"  Using block size: {block_size} bytes (DXGI_FORMAT: {OVOTextureFlipper.DXGI_FORMAT.get(dxgi_format)})",category="", indent=1)
258                else:
259                    raise ValueError(f"DXGI_FORMAT non supportato: {dxgi_format}")
260            else:
261                # Use Four CC for other formats
262                if four_cc in OVOTextureFlipper.BLOCK_SIZE:
263                    block_size = OVOTextureFlipper.BLOCK_SIZE[four_cc]
264                    log(f"  Using block size: {block_size} bytes for format: {four_cc.decode('ascii', errors='replace')}",category="", indent=1)
265                else:
266                    # For uncompressed formats or other formats, try to determine the size basing on pitch
267                    if flags & OVOTextureFlipper.DDSD_PITCH:
268                        raise ValueError(f"Formato non compresso con pitch non supportato")
269                    elif flags & OVOTextureFlipper.DDSD_LINEARSIZE:
270                        # Try to compute size by linear 
271                        linear_size = struct.unpack('<I', data[OVOTextureFlipper.PITCH_OR_LINEAR_SIZE_OFFSET:OVOTextureFlipper.PITCH_OR_LINEAR_SIZE_OFFSET+4])[0]
272                        width_blocks = (width + 3) // 4
273                        height_blocks = (height + 3) // 4
274                        total_blocks = width_blocks * height_blocks
275                        if total_blocks > 0:
276                            estimated_block_size = linear_size / total_blocks
277                            if estimated_block_size <= 12:
278                                block_size = 8
279                            else:
280                                block_size = 16
281                            log(f"  Estimated block size: {estimated_block_size:.2f}  rounded to {block_size}", category="WARNING", indent=1)
282                        else:
283                            raise ValueError(f"Impossibile determinare la dimensione del blocco")
284                    else:
285                        raise ValueError(f"Formato non supportato: {four_cc}")
286
287            # Write new file
288            with open(temp_file, 'wb') as f:
289            
290                # Write header
291                f.write(data[:header_size])
292
293                pos = header_size
294
295                # For each mipmap
296                current_width, current_height = width, height
297                for level in range(mipmap_count):
298
299                    # Block size
300                    width_blocks = max(1, (current_width + 3) // 4)
301                    height_blocks = max(1, (current_height + 3) // 4)
302
303                    # Level dimension
304                    mipmap_size = width_blocks * height_blocks * block_size
305
306                    # Ensure all data is available
307                    if pos + mipmap_size > len(data):
308                        log(f"  WARNING: Mipmap {level} data appears incomplete", category="WARNING", indent=1)
309                        # copy data and exit
310                        f.write(data[pos:])
311                        break
312
313                    # Extract data from mipmap
314                    mipmap_data = data[pos:pos+mipmap_size]
315
316                    # Calculate row length in block
317                    row_size = width_blocks * block_size
318
319                    # Divides data in rows
320                    flipped_rows = []
321                    for row in range(height_blocks): #From top to bottom
322                        start = row * row_size
323                        end   = start + row_size
324                        row_data = mipmap_data[start:end]
325                        new_row  = bytearray()
326
327                        # Iterate through all row blocks
328                        for col in range(width_blocks):
329                            b_start = col * block_size
330                            b_end   = b_start + block_size
331                            blk = row_data[b_start:b_end]
332
333                            # Flip block pixels depending on block size and type
334                            if block_size == 8:
335                                blk = OVOTextureFlipper._flip_bc1_block(blk)
336                            else:  # 16 byte
337                                if four_cc in (OVOTextureFlipper.DXT5_FOURCC,) or \
338                                   dxgi_format in (77, 78, 79):
339                                    blk = OVOTextureFlipper._flip_bc3_block(blk)
340                                elif four_cc in (OVOTextureFlipper.BC5_FOURCC,
341                                                  OVOTextureFlipper.BC5U_FOURCC,
342                                                  OVOTextureFlipper.BC5S_FOURCC) or \
343                                     dxgi_format in (83, 84, 85):
344                                    blk = OVOTextureFlipper._flip_bc5_block(blk)
345                                
346                            new_row.extend(blk)
347
348                        flipped_rows.append(bytes(new_row))
349
350                    # Flip block orders
351                    flipped_mipmap = b''.join(reversed(flipped_rows))
352
353                    # Write flipped mipmap
354                    f.write(flipped_mipmap)
355
356                    # Detailed log only for first level:
357                    if level == 0:
358                        log(f"  Base mipmap: {width_blocks}x{height_blocks} blocks ({row_size} bytes/row)",category="", indent=1)
359
360                    # Update cursor position to swith to the next mipmap level
361                    pos += mipmap_size
362
363                    # Divide by 2 dimension (mipmaps size are always / 2, eg. first original, second original/2, third original/4...)
364                    current_width = max(1, current_width // 2)
365                    current_height = max(1, current_height // 2)
366
367                # If there are additional data copy them
368                if pos < len(data):
369                    remaining_data = data[pos:]
370                    f.write(remaining_data)
371                    log(f"  Copied {len(remaining_data)} extra bytes after mipmaps", category="", indent=1)
372
373            # If we need to overwrite the orignal file, copy the flipped image over
374            if temp_file != final_output:
375                shutil.move(temp_file, final_output)
376
377            log(f"Texture successfully flipped: {final_output}", category="")
378            return final_output
379
380        except (IOError, ValueError) as e:
381            log(f"ERROR during texture flipping: {str(e)}", category="ERROR")
382
383            if temp_file != input_path and output_path is None:
384                if os.path.exists(temp_file):
385                    try:
386                        os.remove(temp_file)
387                    except:
388                        pass
389            return input_path
390        except Exception as e:
391            log(f"UNEXPECTED ERROR during texture flip: {str(e)}", category="ERROR")
392
393            return input_path
