 34    def __init__(self, export_path, use_legacy_compression=True, flip_textures=True):
 35        """
 36        Initializes the texture manager.
 37
 38        Args:
 39            export_path (str): Path of the output OVO file
 40            use_legacy_compression (bool): If True, uses S3TC compression (DXT1/DXT5),
 41                                           otherwise uses BPTC (BC7)
 42            flip_textures (bool): If True, flips textures vertically during export
 43        """
 44        self.export_directory = os.path.dirname(export_path)
 45        self.use_legacy_compression = use_legacy_compression
 46        self.flip_textures = flip_textures
 47        self.addon_directory = os.path.dirname(os.path.abspath(__file__))
 48        self.processed_textures = {}
 49
 50        # Create export directory if it doesn't exist
 51        if not os.path.exists(self.export_directory):
 52            try:
 53                os.makedirs(self.export_directory)
 54                log(f"[OVOTextureManager] Created export directory: {self.export_directory}", category="TEXTURE")
 55            except Exception as e:
 56                log(f"[OVOTextureManager] WARNING: Could not create export directory: {str(e)}", category="WARNING")

 61    def copy_texture_without_compression(self, input_path, output_name=None):
 62        """
 63        Copies a texture without compression when compression is not available.
 64
 65        Args:
 66            input_path (str): Path of the original texture
 67            output_name (str, optional): Output file name
 68
 69        Returns:
 70            str: Name of the copied file or "[none]" in case of error
 71        """
 72        if output_name is None:
 73            output_name = os.path.basename(input_path)
 74
 75        # Destination in the same folder as the OVO export
 76        output_path = os.path.join(self.export_directory, output_name)
 77
 78        try:
 79            # Copy the texture
 80            shutil.copy2(input_path, output_path)
 81            log(f"[OVOTextureManager] Texture copied without compression: '{output_name}'", category="TEXTURE")
 82
 83            # If flipping is enabled and it's a DDS file, try to flip it
 84            if self.flip_textures and output_path.lower().endswith('.dds'):
 85                if OVOTextureFlipper.is_dds_file(output_path):
 86                    try:
 87                        flipped_path = output_path
 88                        OVOTextureFlipper.flip_dds_texture(output_path, flipped_path)
 89                        log(f"[OVOTextureManager] Texture flipped in place: '{output_name}'", category="TEXTURE")
 90                    except Exception as e:
 91                        log(f"[OVOTextureManager] WARNING: Failed to flip texture: {str(e)}", category="WARNING")
 92
 93            return output_name
 94        except Exception as e:
 95            log(f"[OVOTextureManager] ERROR: Failed to copy texture: {str(e)}", category="ERROR")
 96            return "[none]"

101    def compress_texture_to_dds(self, input_path, output_path=None, isAlbedo=False):
102        """
103        Compresses a texture to DDS format using the appropriate compressor for the platform.
104
105        Args:
106            input_path (str): Path of the original texture
107            output_path (str, optional): Output path for the DDS file
108            isAlbedo (bool): If True, it's a texture with alpha channel, otherwise it's a normal map or other
109
110        Returns:
111            tuple: (bool, str) Indicates if compression was successful and the path of the compressed file
112        """
113        if not os.path.exists(input_path):
114            log(f"[OVOTextureManager] ERROR: Input texture does not exist: '{input_path}'", category="ERROR")
115            return False, None
116
117        # Check if the texture has already been processed
118        if input_path in self.processed_textures:
119            log(f"[OVOTextureManager] Using cached texture: '{self.processed_textures[input_path]}'", category="TEXTURE")
120            return True, self.processed_textures[input_path]
121
122        if output_path is None:
123            output_path = os.path.splitext(input_path)[0] + ".dds"
124
125        log(f"[OVOTextureManager] Converting texture: '{os.path.basename(input_path)}'", category="TEXTURE")
126        log(f"- Target: '{os.path.basename(output_path)}'", category="TEXTURE", indent=1)
127        log(f"- Type: {'Albedo' if isAlbedo else 'Non-Albedo'}", category="TEXTURE", indent=1)
128        log(f"- Using legacy compression: {self.use_legacy_compression}", category="TEXTURE", indent=1)
129        log(f"- Flip textures: {self.flip_textures}", category="TEXTURE", indent=1)
130
131        # Determine operating system
132        system = platform.system()
133
134        # Determine format based on use_legacy_compression flag and texture type
135        format = "dxt1"  # Default format
136        try:
137            if isAlbedo:
138                try:
139                    # Load image as binary bytes array
140                    with open(input_path, 'rb') as f:
141                        binary_data = f.read()
142
143                    # Convert to a NumPy array of uint8
144                    binary_array = np.frombuffer(binary_data, dtype=np.uint8)
145
146                    if len(binary_array) % 4 == 0:
147                        pixel_count = len(binary_array) // 4
148                        arr = binary_array.reshape(pixel_count, 4)
149
150                        def has_alpha_channel(arr):
151                            """Checks if the image has a significant alpha channel."""
152                            alpha_channel = arr[:, 3]
153                            return not np.all(alpha_channel == 255)
154
155                        # Choose format based on compression type and alpha presence
156                        if self.use_legacy_compression:
157                            format = "dxt5" if has_alpha_channel(arr) else "dxt1"
158                            log(f"- Alpha channel detected: {has_alpha_channel(arr)}", category="TEXTURE", indent=1)
159                        else:
160                            format = "bc7"  # BC7 handles both with and without alpha
161                    else:
162                        # Otherwise fallback to format without alpha
163                        format = "dxt1" if self.use_legacy_compression else "bc7"
164
165                except Exception as e:
166                    log(f"WARNING: Image analysis failed: {str(e)}", category="WARNING")
167                    log(f"- Using default format for albedo: {'dxt1' if self.use_legacy_compression else 'bc7'}",category="WARNING", indent=1)
168                    format = "dxt1" if self.use_legacy_compression else "bc7"
169            else:
170                format = "bc5"  # BC5 for normal maps
171        except Exception as e:
172            log(f"[OVOTextureManager] WARNING: Format detection failed: {str(e)}", category="WARNING")
173            log(f"- Using fallback format: dxt1", category="WARNING", indent=1)
174
175        log(f"- Selected format: {format.upper()}", category="TEXTURE", indent=1)
176
177        # Get appropriate executable path for the platform
178        result = self._compress_texture_for_platform(system, input_path, output_path, format)
179
180        if result[0]:
181            compressed_path = result[1]
182
183            if self.flip_textures:
184                log(f"- Flipping texture vertically", category="TEXTURE", indent=1)
185                try:
186                    # Use same path to overwrite the original file
187                    flipped_path = OVOTextureFlipper.flip_dds_texture(compressed_path, compressed_path)
188                    log(f"- Texture flipped successfully", category="TEXTURE", indent=1)
189                except Exception as e:
190                    log(f"WARNING: Failed to flip texture: {str(e)}", category="WARNING", indent=1)
191                    log(f"- Original unflipped texture will be used", category="WARNING", indent=1)
192
193            # Add to cache
194            self.processed_textures[input_path] = compressed_path
195            log(f"Compression successful: '{os.path.basename(compressed_path)}'", category="TEXTURE")
196            return True, compressed_path
197        else:
198            log(f"[OVOTextureManager] Compression failed for '{os.path.basename(input_path)}'", category="WARNING")
199            log(f"- Falling back to copy without compression", category="WARNING", indent=1)
200
201            # Fallback to plain copy if compression fails
202            output_name = os.path.basename(output_path)
203            copied_name = self.copy_texture_without_compression(input_path, output_name)
204
205            if copied_name != "[none]":
206                copied_path = os.path.join(self.export_directory, copied_name)
207                self.processed_textures[input_path] = copied_path
208                return True, copied_path
209
210            return False, None

215    def _compress_texture_for_platform(self, system, input_path, output_path, format):
216        """
217        Executes the texture compression for the specified platform.
218
219        Args:
220            system (str): Operating system
221            input_path (str): Path of the original texture
222            output_path (str): Output path for the DDS file
223            format (str): Compression format
224
225        Returns:
226            tuple: (bool, str) Indicates if compression was successful and the path of the compressed file
227        """
228        log(f"Compressing for platform: {system}", category="")
229
230        # Make sure output directory exists
231        output_dir = os.path.dirname(output_path)
232        if not os.path.exists(output_dir):
233            try:
234                os.makedirs(output_dir)
235                log(f"Created output directory: {output_dir}", category="", indent=1)
236            except Exception as e:
237                log(f"ERROR: Failed to create output directory: {str(e)}", category="ERROR", indent=1)
238                return False, None
239
240        # Compression on macOS
241        if system == "Darwin":  # macOS
242            compressor_path = os.path.join(self.addon_directory, "bin", "dds_compress")
243
244            # Check that the executable exists
245            if not os.path.exists(compressor_path):
246                log(f"[OVOTextureManager] ERROR: Executable not found at '{compressor_path}'", category="ERROR", indent=1)
247                return False, None
248
249            # On macOS, make the executable executable
250            os.chmod(compressor_path, 0o755)
251
252            # Run the compressor
253            try:
254                cmd = [compressor_path, input_path, output_path, format.lower()]
255
256                result = subprocess.run(
257                    cmd,
258                    check=True,
259                    capture_output=True,
260                    text=True
261                )
262
263                if result.returncode == 0:
264                    log("Compression successful", category="TEXTURE", indent=1)
265                    return True, output_path
266                else:
267                    log(f"Compression error: {result.stderr}", category="ERROR", indent=1)
268                    return False, None
269            except Exception as e:
270                log(f"ERROR: Failed to execute compressor: {str(e)}", category="ERROR", indent=1)
271                return False, None
272
273        # Compression on Windows
274        elif system == "Windows":
275            compressor_path = os.path.join(self.addon_directory, "bin", "CompressonatorCLI.exe")
276
277            # Check that the executable exists
278            if not os.path.exists(compressor_path):
279                log(f"ERROR: Compressonator CLI not found at '{compressor_path}'", category="ERROR", indent=1)
280                log("Download Compressonator from https://github.com/GPUOpen-Tools/Compressonator/releases",category="TEXTURE", indent=2)
281                log("Place CompressonatorCLI.exe in the bin folder of the addon", category="TEXTURE", indent=2)
282                return False, None
283
284            # Map formats to compressonator format
285            format_map = {
286                "dxt1": "BC1",
287                "dxt5": "BC3",
288                "bc5": "BC5",
289                "bc7": "BC7"
290            }
291
292            comp_format = format_map.get(format.lower(), "BC7")
293
294            # Build command for Compressonator
295            cmd = [
296                compressor_path,
297                "-fd", comp_format,
298                input_path,
299                output_path
300            ]
301
302            try:
303                result = subprocess.run(
304                    cmd,
305                    check=True,
306                    capture_output=True,
307                    text=True
308                )
309
310                if result.returncode == 0:
311                    log("Compression successful", category="TEXTURE", indent=1)
312                    return True, output_path
313                else:
314                    log(f"Compressonator error: {result.stderr}", category="ERROR", indent=1)
315                    return False, None
316            except Exception as e:
317                log(f"ERROR: Failed to execute Compressonator: {str(e)}", category="ERROR", indent=1)
318                return False, None
319
320        # Support for Linux
321        elif system == "Linux":
322            compressor_path = os.path.join(self.addon_directory, "bin", "dds_compress_linux")
323
324            if not os.path.exists(compressor_path):
325                log(f"[OVOTextureManager] ERROR: Compressor not found for Linux at '{compressor_path}'", category="ERROR", indent=1)
326                return False, None
327
328            # For Linux, make the executable executable
329            os.chmod(compressor_path, 0o755)
330
331            # Map formats to compressonator format for Linux
332            format_map = {
333                "dxt1": "BC1",
334                "dxt5": "BC3",
335                "bc5": "BC5",
336                "bc7": "BC7"
337            }
338
339            comp_format = format_map.get(format.lower(), "BC7")
340
341            # Command for Linux Compressonator
342            cmd = [
343                compressor_path,
344                "-fd", comp_format,
345                input_path,
346                output_path
347            ]
348
349            try:
350                result = subprocess.run(
351                    cmd,
352                    check=True,
353                    capture_output=True,
354                    text=True
355                )
356
357                if result.returncode == 0:
358                    log("Compression successful", category="TEXTURE", indent=1)
359                    return True, output_path
360                else:
361                    log(f"Compression error: {result.stderr}", category="ERROR", indent=1)
362                    return False, None
363            except Exception as e:
364                log(f"ERROR: Failed to execute compressor: {str(e)}", category="ERROR", indent=1)
365                return False, None
366
367        else:
368            log(f"[OVOTextureManager] ERROR: Unsupported operating system: {system}", category="ERROR", indent=1)
369            return False, None

374    def trace_to_image_node(self, input_item, isAlbedo=False):
375        """
376        Recursive function to trace up the node chain to the image texture node,
377        also handling intermediate nodes (e.g. Bright/Contrast).
378
379        Args:
380            input_item: Input socket or node from which to start the search
381            isAlbedo (bool): If True, the texture is an albedo/color texture
382
383        Returns:
384            str: Name of the processed texture or "[none]" if not found
385        """
386        # If the element doesn't have the is_linked attribute, it might be a node;
387        # in this case, take the "Color" socket if present.
388        if not hasattr(input_item, "is_linked"):
389            input_item = input_item.inputs.get("Color")
390            if not input_item:
391                return "[none]"
392
393        # If the socket is not linked, exit
394        if not input_item or not input_item.is_linked:
395            return "[none]"
396
397        # Get the source socket and related node
398        from_socket = input_item.links[0].from_socket
399        from_node = from_socket.node
400        log(f"[OVOTextureManager] Tracing from input: '{input_item.name}'", category="TEXTURE", indent=1)
401        log(f"Connected node type: {type(from_node).__name__}", category="TEXTURE", indent=2)
402
403        # Base case: if the node is an Image Texture, save the texture
404        if isinstance(from_node, bpy.types.ShaderNodeTexImage):
405            log("Found Image Texture node directly", category="TEXTURE", indent=2)
406            if from_node.image:
407                image = from_node.image
408                log(f"Image name: '{image.name}'", category="TEXTURE", indent=2)
409                # If the image is packed, save it to the output folder
410                if image.packed_file:
411                    texture_filename = image.name
412                    output_path = os.path.join(self.export_directory, texture_filename)
413                    try:
414                        image.save_render(output_path)
415                        log(f"Saved packed image to: '{output_path}'", category="TEXTURE", indent=2)
416                        # Now compress the texture to DDS
417                        dds_output = os.path.splitext(output_path)[0] + ".dds"
418                        success, dds_path = self.compress_texture_to_dds(
419                            output_path,
420                            dds_output,
421                            isAlbedo=isAlbedo
422                        )
423                        if success:
424                            os.remove(output_path)  # Remove original file after compression
425                            texture_name_dds = os.path.splitext(os.path.basename(dds_path))[0] + ".dds"
426                            return texture_name_dds
427                        else:
428                            # Fallback: use uncompressed texture
429                            log("Compression failed, using uncompressed texture", category="WARNING", indent=2)
430                            return self.copy_texture_without_compression(output_path)
431                    except Exception as e:
432                        log(f"ERROR: Failed to export texture '{texture_filename}': {e}", category="ERROR", indent=2)
433                        return "[none]"
434
435                # For the case of images with filepath:
436                elif image.filepath:
437                    source_path = bpy.path.abspath(image.filepath)
438                    if os.path.exists(source_path):
439                        texture_filename = os.path.basename(source_path)
440                        output_path = os.path.join(self.export_directory, texture_filename)
441                        try:
442                            log(f"Using image from filepath: '{source_path}'", category="TEXTURE", indent=2)
443                            dds_output = os.path.splitext(output_path)[0] + ".dds"
444                            success, dds_path = self.compress_texture_to_dds(
445                                source_path,
446                                dds_output,
447                                isAlbedo=isAlbedo
448                            )
449                            if success:
450                                texture_name_dds = os.path.splitext(os.path.basename(dds_path))[0] + ".dds"
451                                return texture_name_dds
452                            else:
453                                # Fallback: use uncompressed texture
454                                log("Compression failed, using uncompressed texture", category="WARNING", indent=2)
455                                return self.copy_texture_without_compression(source_path)
456                        except Exception as e:
457                            log(f"ERROR: Failed to compress texture: {e}", category="ERROR", indent=2)
458                            # Fallback: use uncompressed texture
459                            return self.copy_texture_without_compression(source_path)
460                    else:
461                        log(f"ERROR: Image filepath does not exist: '{source_path}'", category="ERROR", indent=2)
462            return "[none]"
463
464        # If the node is not an Image Texture, try to trace through the "Color" socket
465        if hasattr(from_node, 'inputs'):
466            color_input = from_node.inputs.get('Color')
467            if color_input and color_input.is_linked:
468                log("Following 'Color' input connection...", category="TEXTURE", indent=2)
469                return self.trace_to_image_node(color_input, isAlbedo=isAlbedo)
470            # If there's no "Color", try all linked inputs
471            for inp in from_node.inputs:
472                if inp.is_linked:
473                    log(f"Following '{inp.name}' input connection...", category="TEXTURE", indent=2)
474                    result = self.trace_to_image_node(inp, isAlbedo=isAlbedo)
475                    if result != "[none]":
476                        return result
477
478                log("No image found in node chain", category="TEXTURE", indent=2)
479                return "[none]"

375    def trace_to_image_node(self, input_item, isAlbedo=False):
376        """
377        Recursive function to trace up the node chain to the image texture node,
378        also handling intermediate nodes (e.g. Bright/Contrast).
379
380        Args:
381            input_item: Input socket or node from which to start the search
382            isAlbedo (bool): If True, the texture is an albedo/color texture
383
384        Returns:
385            str: Name of the processed texture or "[none]" if not found
386        """
387        # If the element doesn't have the is_linked attribute, it might be a node;
388        # in this case, take the "Color" socket if present.
389        if not hasattr(input_item, "is_linked"):
390            input_item = input_item.inputs.get("Color")
391            if not input_item:
392                return "[none]"
393
394        # If the socket is not linked, exit
395        if not input_item or not input_item.is_linked:
396            return "[none]"
397
398        # Get the source socket and related node
399        from_socket = input_item.links[0].from_socket
400        from_node = from_socket.node
401        log(f"[OVOTextureManager] Tracing from input: '{input_item.name}'", category="TEXTURE", indent=1)
402        log(f"Connected node type: {type(from_node).__name__}", category="TEXTURE", indent=2)
403
404        # Base case: if the node is an Image Texture, save the texture
405        if isinstance(from_node, bpy.types.ShaderNodeTexImage):
406            log("Found Image Texture node directly", category="TEXTURE", indent=2)
407            if from_node.image:
408                image = from_node.image
409                log(f"Image name: '{image.name}'", category="TEXTURE", indent=2)
410                # If the image is packed, save it to the output folder
411                if image.packed_file:
412                    texture_filename = image.name
413                    output_path = os.path.join(self.export_directory, texture_filename)
414                    try:
415                        image.save_render(output_path)
416                        log(f"Saved packed image to: '{output_path}'", category="TEXTURE", indent=2)
417                        # Now compress the texture to DDS
418                        dds_output = os.path.splitext(output_path)[0] + ".dds"
419                        success, dds_path = self.compress_texture_to_dds(
420                            output_path,
421                            dds_output,
422                            isAlbedo=isAlbedo
423                        )
424                        if success:
425                            os.remove(output_path)  # Remove original file after compression
426                            texture_name_dds = os.path.splitext(os.path.basename(dds_path))[0] + ".dds"
427                            return texture_name_dds
428                        else:
429                            # Fallback: use uncompressed texture
430                            log("Compression failed, using uncompressed texture", category="WARNING", indent=2)
431                            return self.copy_texture_without_compression(output_path)
432                    except Exception as e:
433                        log(f"ERROR: Failed to export texture '{texture_filename}': {e}", category="ERROR", indent=2)
434                        return "[none]"
435
436                # For the case of images with filepath:
437                elif image.filepath:
438                    source_path = bpy.path.abspath(image.filepath)
439                    if os.path.exists(source_path):
440                        texture_filename = os.path.basename(source_path)
441                        output_path = os.path.join(self.export_directory, texture_filename)
442                        try:
443                            log(f"Using image from filepath: '{source_path}'", category="TEXTURE", indent=2)
444                            dds_output = os.path.splitext(output_path)[0] + ".dds"
445                            success, dds_path = self.compress_texture_to_dds(
446                                source_path,
447                                dds_output,
448                                isAlbedo=isAlbedo
449                            )
450                            if success:
451                                texture_name_dds = os.path.splitext(os.path.basename(dds_path))[0] + ".dds"
452                                return texture_name_dds
453                            else:
454                                # Fallback: use uncompressed texture
455                                log("Compression failed, using uncompressed texture", category="WARNING", indent=2)
456                                return self.copy_texture_without_compression(source_path)
457                        except Exception as e:
458                            log(f"ERROR: Failed to compress texture: {e}", category="ERROR", indent=2)
459                            # Fallback: use uncompressed texture
460                            return self.copy_texture_without_compression(source_path)
461                    else:
462                        log(f"ERROR: Image filepath does not exist: '{source_path}'", category="ERROR", indent=2)
463            return "[none]"
464
465        # If the node is not an Image Texture, try to trace through the "Color" socket
466        if hasattr(from_node, 'inputs'):
467            color_input = from_node.inputs.get('Color')
468            if color_input and color_input.is_linked:
469                log("Following 'Color' input connection...", category="TEXTURE", indent=2)
470                return self.trace_to_image_node(color_input, isAlbedo=isAlbedo)
471            # If there's no "Color", try all linked inputs
472            for inp in from_node.inputs:
473                if inp.is_linked:
474                    log(f"Following '{inp.name}' input connection...", category="TEXTURE", indent=2)
475                    result = self.trace_to_image_node(inp, isAlbedo=isAlbedo)
476                    if result != "[none]":
477                        return result
478
479                log("No image found in node chain", category="TEXTURE", indent=2)
480                return "[none]"

416                        log(f"Saved packed image to: '{output_path}'", category="TEXTURE", indent=2)
417                        # Now compress the texture to DDS
418                        dds_output = os.path.splitext(output_path)[0] + ".dds"
419                        success, dds_path = self.compress_texture_to_dds(
420                            output_path,
421                            dds_output,
422                            isAlbedo=isAlbedo
423                        )
424                        if success:
425                            os.remove(output_path)  # Remove original file after compression
426                            texture_name_dds = os.path.splitext(os.path.basename(dds_path))[0] + ".dds"
427                            return texture_name_dds
428                        else:
429                            # Fallback: use uncompressed texture
430                            log("Compression failed, using uncompressed texture", category="WARNING", indent=2)
431                            return self.copy_texture_without_compression(output_path)
432                    except Exception as e:
433                        log(f"ERROR: Failed to export texture '{texture_filename}': {e}", category="ERROR", indent=2)
434                        return "[none]"
435
436                # For the case of images with filepath:
437                elif image.filepath:
438                    source_path = bpy.path.abspath(image.filepath)
439                    if os.path.exists(source_path):
440                        texture_filename = os.path.basename(source_path)
441                        output_path = os.path.join(self.export_directory, texture_filename)
442                        try:
443                            log(f"Using image from filepath: '{source_path}'", category="TEXTURE", indent=2)
444                            dds_output = os.path.splitext(output_path)[0] + ".dds"
445                            success, dds_path = self.compress_texture_to_dds(
446                                source_path,
447                                dds_output,
448                                isAlbedo=isAlbedo
449                            )
450                            if success:
451                                texture_name_dds = os.path.splitext(os.path.basename(dds_path))[0] + ".dds"
452                                return texture_name_dds
453                            else:
454                                # Fallback: use uncompressed texture
455                                log("Compression failed, using uncompressed texture", category="WARNING", indent=2)
456                                return self.copy_texture_without_compression(source_path)
457                        except Exception as e:
458                            log(f"ERROR: Failed to compress texture: {e}", category="ERROR", indent=2)
459                            # Fallback: use uncompressed texture
460                            return self.copy_texture_without_compression(source_path)
461                    else:
462                        log(f"ERROR: Image filepath does not exist: '{source_path}'", category="ERROR", indent=2)
463            return "[none]"
464
465        # If the node is not an Image Texture, try to trace through the "Color" socket
466        if hasattr(from_node, 'inputs'):
467            color_input = from_node.inputs.get('Color')
468            if color_input and color_input.is_linked:
469                log("Following 'Color' input connection...", category="TEXTURE", indent=2)
470                return self.trace_to_image_node(color_input, isAlbedo=isAlbedo)
471            # If there's no "Color", try all linked inputs
472            for inp in from_node.inputs:
473                if inp.is_linked:
474                    log(f"Following '{inp.name}' input connection...", category="TEXTURE", indent=2)
475                    result = self.trace_to_image_node(inp, isAlbedo=isAlbedo)
476                    if result != "[none]":
477                        return result
478
479                log("No image found in node chain", category="TEXTURE", indent=2)
480                return "[none]"

Recursive function to trace up the node chain to the image texture node, also handling intermediate nodes (e.g. Bright/Contrast).

Args: input_item: Input socket or node from which to start the search isAlbedo (bool): If True, the texture is an albedo/color texture

Returns: str: Name of the processed texture or "[none]" if not found
